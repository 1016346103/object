 14 this.camelCase = function(str) {

 16 		return match.charAt(1).toUpperCase();
 17 	});
 18 };
 19 
 20 this.hyphenate = function(str) {

 22 		return ('-' + match.charAt(0).toLowerCase());
 23 	});
 24 };
 25 
 26 this.capitalize = function(str) {

 28 		return match.toUpperCase();
 29 	});

 32 this.trim = function(str) {

 48 /**
 49  * 将对象转换为querystring
 50  * 来自 mootools
 51  */
 52 this.toQueryString = function(object) {
 53 	var queryString = [];
 54 
 55 	for (var key in object) {
 56 		var value = object[key];
 57 
 58 		var result;
 59 
 60 		if (value && value.constructor === Array) {
 61 			var qs = {};
 62 			value.forEach(function(val, i) {
 63 				qs[i] = val;
 64 			});
 65 
 66 			result = arguments.callee(qs, key);
 67 		} else if (typeof value == 'object') {
 68 			result = arguments.callee(value, key);
 69 		} else {
 70 			result = key + '=' + encodeURIComponent(value);
 71 		}
 72 
 73 		if (value !== null) queryString.push(result);
 74 	}
 75 
 76 	return queryString.join('&');
 77 };
