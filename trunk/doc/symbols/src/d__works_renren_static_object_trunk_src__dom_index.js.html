 82 /**
 83  * 包装一个元素，使其拥有相应的Element包装成员
 84  * 比如 div 会使用 Element 进行包装
 85  * form 会使用 FormElement 进行包装
 86  * input / select 等会使用 FormItemElement 进行包装
 87  * 包装后的节点成员请参照相应的包装类成员
 88  * @function
 89  * @name dom.wrap
 90  * @param node 一个原生节点
 91  */
 92 var wrap = this.wrap = function(node) {
 93 	if (!node) return null;
 94 
 95 	if (Array.isArray(node)) {
 96 		return new Elements(node);
 97 	} else {
 98 		// 已经wrap过了
 99 		if (node._wrapped) return node;
100 
101 		var wrapper;
102 		if (node === window) {
103 			wrapper = Window;
104 		} else if (node === window.document) {
105 			wrapper = Document;
106 		} else if (node.nodeType === 1) {
107 			wrapper = getWrapper(node.tagName);
108 		} else {
109 			return node;
110 		}
111 
112 		// 尽早的设置_wrapped，因为在wrapper的initialize中可能出现递归调用（FormElement/FormItemElement）
113 		node._wrapped = true;
114 
115 		$uid(node);
116 
117 		Class.inject(wrapper, node);
118 
119 		return node;
120 	}
121 };

123 /**
124  * 通过selector获取context作用域下的节点集合
125  * dom.Elements包装后的节点数组拥有相应最小Element的统一调用方法
126  * 比如 forms = dom.getElements('form'); 'send' in forms // true
127  * @function
128  * @name dom.getElements
129  * @param selector 一个css selector
130  * @param context 一个节点
131  * @returns {dom.Elements}
132  */
133 var getElements = this.getElements = function(selector, context) {
134 	if (!context) context = document;
135 
136 	// 解析成Slick Selector对象
137 	var parsed = Slick.parse(selector);
138 
139 	// Slick在面对自定义标签时各种不靠谱，换用sizzle
140 	var eles = Sizzle(selector, context);
141 
142 	// 这里通过分析selector的最后一个部分的tagName，来确定这批eles的wrapper
143 	// 例如selector是 div form.xxx 则wrapper是 FormElement
144 	// 例如selector是 div .xxx 则wrapper是 Element
145 	// 例如selector是 div select.xxx, div.input.xxx 则wrapper是 FormItemElement
146 
147 	var wrapper, part;
148 	// 绝大部分情况都是length=0，只有1个selector，保证其性能
149 	if (parsed.expressions.length == 1) {
150 		part = parsed.expressions[0];
151 		wrapper = getWrapper(part[part.length - 1].tag);
152 
153 	// 由多个selector组成，比如 div select.xxx, div.input.xxx，要保证这种能取到 FormItemElement
154 	} else {
155 		// 通过生成每个selector wrapper的继承链，不断的生成当前selector和上一个selector的继承链的相同部分
156 		// 最后的chain的最后一个元素，既是公用wrapper
157 		for (var i = 0, chain, previousChain; i < parsed.expressions.length; i++) {
158 			part = parsed.expressions[i];
159 			wrapper = getWrapper(part[part.length - 1].tag);
160 
161 			// 当前selector最后元素的wrapper chain
162 			// slice(0, -1) 过滤掉Element继承的 Attribute 类
163 			chain = Class.getChain(wrapper).slice(0, -1).reverse();
164 			if (previousChain) {
165 				chain = getCommon(chain, previousChain);
166 			}
167 			// 如果相同部分length=1，则代表找到Element类了，可以停止继续搜索
168 			if (chain.length == 1) break;
169 			previousChain = chain;
170 		}
171 		wrapper = chain[chain.length - 1];
172 	}
173 
174 	return new Elements(eles, wrapper);
175 };

207 	var js = [];

210 			if (ele.childNodes[i].nodeType === 1) {

212 			}

223 	for (i = 0; i < js.length; i++) {
224 		arr.push(js[i]);
225 	}
226 
227 	arr.forEach(function(s, i) {
228 		if (s.src) {
229 			// TODO

232 			var inner_js = '__inner_js_out_put = [];\n';
233 			inner_js += s.innerHTML.replace( /document\.write/g, '__inner_js_out_put.push' );

235 			if (__inner_js_out_put.length !== 0) {
