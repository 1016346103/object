1 /**
  2  * @namespace
  3  * @name options
  4  */
  5 object.add('options', /**@lends options*/ function(exports) {
  6 
  7 /**
  8  * 这个类辅助这种参数传递方式的实现：
  9  * callFunc({
 10  *	param1: someValue1,
 11  *	param2: someValue2
 12  * })
 13  * 在声明函数时，通过：
 14  * var opts = new ns.Arguments(opts, {
 15  *	param1: 1,
 16  *	param2: 2
 17  * });
 18  * 来设定默认值，没有设置过默认值的成员不会输出
 19  * @class
 20  */
 21 this.Arguments = new Class(function() {
 22 
 23 	/**
 24 	 * @param defaults 默认值列表
 25 	 * @param opts 参数列表
 26 	 */
 27 	this.initialize = function(self, defaults, opts) {
 28 		if (opts == undefined) opts = {};
 29 
 30 		var output = {};
 31 		for (var key in defaults) {
 32 			output[key] = (opts[key] != undefined? opts[key] : defaults[key]);
 33 		}
 34 		return output;
 35 	};
 36 
 37 });
 38 
 39 /**
 40  * 参数
 41  * @class
 42  */
 43 this.Options = new Class({
 44 
 45 	/**
 46 	 * 提供一个实现了 makeOption 接口的“提供者”参数，这样，在 setOption 时会自动根据name获取value，不用手工调用
 47 	 */
 48 	initialize: function(self, provider) {
 49 		if (provider) self._provider = provider;
 50 		self._options = {};
 51 	},
 52 
 53 	setOptions: function(self, options, host) {
 54 		if (!host) host = self._options;
 55 
 56 		for (var i in options) {
 57 			if (host[i] !== undefined) host[i] = options[i];
 58 		}
 59 	},
 60 
 61 	setOption: function(self, name, type, value) {
 62 		if (value !== undefined) {
 63 			self._options[name] = value;
 64 		} else {
 65 			value = self._provider.makeOption(name, type);
 66 			if (value === null) return;
 67 			else self._options[name] = value;
 68 		}
 69 	},
 70 
 71 	getOptions: function(self) {
 72 		return self._options;
 73 	}
 74 
 75 });
 76 
 77 });
