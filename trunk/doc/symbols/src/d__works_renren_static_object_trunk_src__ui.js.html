  7 /**
  8  * 定义sub components
  9  * @name ui.define
 10  */
 11 var define = this.define = function(cls, name, selector, type, single) {
 12 
 13 	var getter = function(self) {
 14 		if (!type) type = Component;
 15 
 16 		if (!self._descriptors[name]) {
 17 			self._descriptors[name] = {
 18 				selector: selector,
 19 				type: type,
 20 				single: single
 21 			};
 22 		}
 23 
 24 		var pname = '_' + name;
 25 		if (single) {
 26 			var node = self._node.getElement(selector);
 27 			if (!node) return null;
 28 			else if (self[pname] === node) return self[name]; // 不是第一次get，且结果和上次相同，避免再次添加事件
 29 
 30 			self._addEventTo(name, node);
 31 			self[pname] = node;
 32 
 33 			return new type(node, self._options[name]);
 34 		} else {
 35 			var nodes = self._node.getElements(selector);
 36 			if (!nodes) {
 37 				self[pname] = new dom.Elements([]);
 38 				return new Components([], type);
 39 			}
 40 
 41 			// 如果已经初始化过，则要确保不会对之前取到过得元素重新执行添加事件
 42 			nodes.forEach(function(node) {
 43 				if (!self[pname] || self[pname].indexOf(node) === -1) {
 44 					self._addEventTo(name, node);
 45 				}
 46 			});
 47 			self[pname] = nodes;
 48 
 49 			return new Components(nodes, type, self._options[name], self);
 50 		}
 51 	};
 52 
 53 	cls[name] = property(getter);
 54 };

 64 /**
 65  * 定义 options
 66  * @name ui.defineOptions
 67  */
 68 var defineOptions = this.defineOptions = function(cls, options) {
 69 	Object.keys(options).forEach(function(name) {
 70 		var pname = '_' + name;
 71 		var methodName = name + '_change';
 72 		cls[name] = property(function(self) {
 73 			if (self[pname] === undefined) {
 74 				self[pname] = options[name];
 75 			}
 76 			return self[pname];
 77 		}, function(self, value) {
 78 			if (self[methodName]) {
 79 				self[methodName](value);
 80 			}
 81 			self[pname] = value;
 82 			self._set(name, value);
 83 			return self[pname];
 84 		});
 85 	});
 86 };

 88 /**
 89  * @class
 90  */
 91 var Element = new Class(function() {
 92 
 93 	Object.keys(dom.Element).forEach(function(name) {
 94 		if (typeof dom.Element[name] === 'function') {
 95 			if (['initialize'].indexOf(name) != -1) return;
 96 
 97 			this[name] = function(self) {
 98 				var args = [self._node];
 99 				var arg;
100 				// 代理方法支持Component参数
101 				for (var i = 1; i < arguments.length; i++) {
102 					arg = arguments[i];
103 					args.push((arg && arg._node)? arg._node : arg);
104 				}
105 				return dom.Element[name].apply(dom.Element, args);
106 			};
107 		}
108 	}, this);
109

113 /**
114  * @class
115  * @name ui.Components
116  */
117 var Components = this.Components = new Class(Array, function() {
118 
119 	/**
120 	 * @param elements wrapped dom elements
121 	 * @param type 这批节点的共有Component类型，默认为Component
122 	 */
123 	this.initialize  = function(self, elements, type, options) {
124 		if (!type) type = Component;
125 
126 		for (var i = 0; i < elements.length; i++) {
127 			self.push(new type(elements[i], options));
128 		}
129 
130 		Object.keys(type).forEach(function(name) {
131 			self[name] = function() {
132 				var element;
133 				var i, arg, args = [];
134 				// 代理方法支持Component参数
135 				for (i = 0; i < arguments.length; i++) {
136 					arg = arguments[i];
137 					args.push((arg && arg._node)? arg._node : arg);
138 				}
139 				for (i = 0; i < self.length; i++) {
140 					element = self[i];
141 					if (typeof element[name] == 'function') {
142 						element[name].apply(self[i], args);
143 					}
144 				}
145 			};
146 		});
147 
148 		self.set = function(key, value) {
149 			for (var i = 0; i < self.length; i++) {
150 				self[i].set(key, value);
151 			}
152 		};
153 
154 		self.get = function(key) {
155 			var result = [];
156 			for (var i = 0; i < self.length; i++) {
157 				result.push(self[i].get(key));
158 			}
159 			return result;
160 		};
161 	};
162 
163 });

166 /**
167  * UI模块基类
168  * @class
169  * @name ui.Component
170  */
171 var Component = this.Component = new Class(function() {
172 
173 	var getConstructor = function(type) {
174 		if (type === 'number') return Number;
175 		else if (type === 'string') return String;
176 		else if (type === 'boolean') return Boolean;
177 	}
178 
179 	Class.mixin(this, Element);
180 
181 	this.initialize = function(self, node, options) {
182 		if (!options) options = {};
183 		self._descriptors = {};
184 		self._components = []; // 建立出来的所有子component的引用
185 		self._rendered = []; // render出来的新元素，会在reset时清空
186 		self._events = self._getEvents(self); // 本class的所有event方法
187 		self._options = self._parseOptions(options);
188 		self._node = node;
189 		self._optionNames = [];
190 
191 		Class.getPropertyNames(self).forEach(function(name) {
192 			// 从dom获取配置
193 			var data = self._node.getData(name.toLowerCase());
194 			if (data) {
195 				var defaultValue = self.get(name);
196 				var value = getConstructor(typeof defaultValue)(data);
197 				self.set(name, value);
198 			// 从options参数获取配置
199 			} else if (self._options[name] && typeof self._options[name] !== 'object') {
200 				self.set(name, self._options[name]);
201 			// 默认配置
202 			} else {
203 				self[name] = self.get(name);
204 
205 				// 是option，不是component
206 				if (!self._descriptors[name]) {
207 					self.set(name, self[name]);
208 				}
209 			}
210 		});
211 	};
212 
213 	/**
214 	 * 解析options为对象
215 	 * {'a.b.c': 1, b: 2} ==> {a: {'b.c': 1}, b: 2}
216 	 */
217 	this._parseOptions = staticmethod(function(options) {
218 		var parsed = {};
219 		Object.keys(options).forEach(function(key) {
220 			var point = key.indexOf('.');
221 			if (point !== -1) {
222 				var first = key.substring(0, point);
223 				if (!parsed[first]) parsed[first] = {};
224 				parsed[first][key.substring(point + 1)] = options[key];
225 			} else {
226 				parsed[key] = options[key];
227 			}
228 		});
229 		return parsed;
230 	});
231 
232 	this._addEventTo = function(self, name, node) {
233 		var events = self._events[name];
234 		if (!events) return;
235 
236 		if (events) events.forEach(function(event) {
237 			node.addEvent(event.name, event.func);
238 		});
239 	};
240 
241 	this.render = function(self, name, data) {
242 		var methodName = 'render' + string.capitalize(name);
243 		var descriptor = self._descriptors[name];
244 		var type = descriptor.type;
245 		if (!self[methodName]) return;
246 		var result = self[methodName](data);
247 		// 如果有返回结果，说明没有使用self.make，而是自己生成了需要的普通node元素，则对返回结果进行一次包装
248 		if (result) {
249 			if (Array.isArray(result)) {
250 				result.forEach(function(node) {
251 					self.registerComponent(name, node, data);
252 				});
253 			} else {
254 				self.registerComponent(name, result, data);
255 			}
256 		}
257 	};
258 
259 	/**
260 	* 根据descriptors的type创建一个component，这一般是在renderXXX方法中进行调用
261 	* @param name
262 	* @param data 模板数据
263 	*/
264 	this.make = function(self, name, data) {
265 		var descriptor = self._descriptors[name];
266 		var template = descriptor.template;
267 		var secName = descriptor.secName;
268 
269 		if (!data) data = {};
270 		var tdata = {};
271 		if (secName) {
272 			tdata[secName] = data;
273 		} else {
274 			tdata = data;
275 		}
276 
277 		var str = string.substitute(template, tdata);
278 		var node = dom.Element.fromString(str).firstChild;
279 		var comp = self.registerComponent(name, node, data);
280 
281 		return comp;
282 	};
283 
284 	/**
285 	 * 根据节点初始化一个component，并放到相应的引用上去。
286 	 */
287 	this.registerComponent = function(self, name, node, data) {
288 		var descriptor = self._descriptors[name];
289 		var type = descriptor.type;
290 		var single = descriptor.single;
291 		var pname = '_' + name;
292 
293 		var options = {};
294 		var extendOptions = self._options[name];
295 
296 		if (extendOptions) {
297 			Object.keys(extendOptions).forEach(function(key) {
298 				options[key] = extendOptions[key];
299 			});
300 		}
301 		if (data) {
302 			Object.keys(data).forEach(function(key) {
303 				options[key] = data[key];
304 			});
305 		}
306 
307 		var comp = new type(node, options);
308 
309 		if (single) {
310 			self[name] = comp;
311 			self[pname] = node;
312 			self._addEventTo(name, node);
313 			self._rendered.push(node);
314 		} else {
315 			self[name].push(comp);
316 			self[pname].push(node);
317 			self._addEventTo(name, node);
318 			self._rendered.push(node);
319 		}
320 
321 		return comp;
322 	};
323 
324 	this.call = function(self, name) {
325 		self._node.fireEvent(name, null, self);
326 		if (!self[name]) throw 'no method named ' + name;
327 		self[name].apply(self, [].slice.call(arguments, 2));
328 	};
329 
330 	this.apply = function(self, name, args) {
331 		self._node.fireEvent(name, null, self);
332 		if (!self[name]) throw 'no method named ' + name;
333 		self[name].apply(self, args);
334 	};
335 
336 	this.setTemplate = function(self, name, template, secName) {
337 		self._descriptors[name].template = template;
338 		self._descriptors[name].secName = secName;
339 	};
340 
341 	/**
342 	 * 渲染一个新的控件
343 	 * @param template 模板字符串
344 	 * @param data 模板数据
345 	 * @param secName 模板片段名称
346 	 */
347 	this.create = classmethod(function(cls, template, data, secName) {
348 		if (!data) data = {};
349 		var tdata = {};
350 		if (secName) {
351 			tdata[secName] = data;
352 		} else {
353 			tdata = data;
354 		}
355 
356 		var str = string.substitute(template, tdata);
357 		var node = dom.Element.fromString(str).firstChild;
358 		return new cls(node);
359 	});
360 
361 	this._getEvents = classmethod(function(cls, self) {
362 		var events = {};
363 
364 		Object.keys(cls).forEach(function(key) {
365 			var match = key.match(/^(_?[a-zA-Z]+)_([a-zA-Z]+)$/);
366 			if (!match) return;
367 			var name = match[1];
368 			var eventName = match[2];
369 
370 			if (!events[name]) events[name] = [];
371 			events[name].push({
372 				name: eventName,
373 				func: self[key].bind(self)
374 			});
375 		});
376 
377 		return events;
378 	});
379 
380 	this.invalid = function(self, msg) {
381 		if (!msg) msg = '输入错误';
382 		alert(msg);
383 	};
384 
385 	this.error = function(self, msg) {
386 		if (!msg) msg = '出错啦！';
387 		alert(msg);
388 	};
389 
390 	this.reset = function(self) {
391 		// 清空所有render进来的新元素
392 		self._rendered.forEach(function(node) {
393 			node.dispose();
394 		});
395 		// 所有子component reset
396 		self._components.forEach(function(name) {
397 			var single = self._descriptors[name].single;
398 			if (single) {
399 				if (self[name] && self[name].reset) self[name].reset();
400 			} else {
401 				if (self[name]) {
402 					self[name].forEach(function(comp) {
403 						comp.reset();
404 					});
405 				}
406 			}
407 		});
408 	};
409 
410 	this.define = staticmethod(define);
411 	this.define1 = staticmethod(define1);
412 	this.defineOptions = staticmethod(defineOptions);
413 
414 });

416 /**
417  * @class
418  * @name ui.ForeNextControl
419  */
420 var ForeNextControl = this.ForeNextControl = new Class(Component, function() {
421 
422 	Component.define(this, 'nextButton', '.nextbutton');
423 	Component.define(this, 'foreButton', '.forebutton');
424 
425 	this.initialize = function(self, node) {
426 		Component.initialize(self, node);
427 
428 		self.loop = false; // 是否循环
429 		self.total = parseInt(self._node.getData('total'));
430 		self.start = parseInt(self._node.getData('start')) || 0;
431 		self.position = self.start;
432 	};
433 
434 	this.nextButton_click = function(self, event) {
435 		if (self.position >= self.total - 1) {
436 			if (self.loop) self.position = -1;
437 			else return;
438 		}
439 		self.call('next');
440 	};
441 
442 	this.foreButton_click = function(self, event) {
443 		if (self.position <= 0) {
444 			if (self.loop) self.position = self.total;
445 			else return;
446 		}
447 		self.call('fore');
448 	};
449 
450 	this.next = function(self) {
451 		self.position++;
452 		self.call('change');
453 	};
454 
455 	this.fore = function(self) {
456 		self.position--;
457 		self.call('change');
458 	};
459 
460 	this.change = function(self) {
461 		self.call('updateTotal');
462 		self.call('updatePosition');
463 	};
464 
465 	this.updatePosition = function(self) {
466 		self._node.getElements('.current').set('innerHTML', self.position + 1); // position是从0开始滴～展示的时候+1
467 	};
468 
469 	this.updateTotal = function(self) {
470 		self._node.getElements('.total').set('innerHTML', self.total);
471 	};
472 
473 });
