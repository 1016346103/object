1 /**
  2  * @namespace
  3  * @name ua
  4  */
  5 object.add('ua', /**@lends ua*/ function($) {
  6 
  7 var o = this.ua = {};
  8 var ua = navigator.userAgent,
  9 	EMPTY = '', MOBILE = 'mobile',
 10 	core = EMPTY, shell = EMPTY, m,
 11 	//o = {
 12 		// browser core type
 13 		//webkit: 0,
 14 		//trident: 0,
 15 		//gecko: 0,
 16 		//presto: 0,
 17 
 18 		// browser type
 19 		//chrome: 0,
 20 		//safari: 0,
 21 		//firefox: 0,
 22 		//ie: 0,
 23 		//opera: 0
 24 
 25 		//mobile: '',
 26 		//core: '',
 27 		//shell: ''
 28 	//},
 29 	numberify = function(s) {
 30 		var c = 0;
 31 		// convert '1.2.3.4' to 1.234
 32 		return parseFloat(s.replace(/\./g, function() {
 33 			return (c++ === 0) ? '.' : '';
 34 		}));
 35 	};
 36 
 37 // WebKit
 38 if ((m = ua.match(/AppleWebKit\/([\d.]*)/)) && m[1]) {
 39 	o[core = 'webkit'] = numberify(m[1]);
 40 
 41 	// Chrome
 42 	if ((m = ua.match(/Chrome\/([\d.]*)/)) && m[1]) {
 43 		o[shell = 'chrome'] = numberify(m[1]);
 44 	}
 45 	// Safari
 46 	else if ((m = ua.match(/\/([\d.]*) Safari/)) && m[1]) {
 47 		o[shell = 'safari'] = numberify(m[1]);
 48 	}
 49 
 50 	// Apple Mobile
 51 	if (/ Mobile\//.test(ua)) {
 52 		o[MOBILE] = 'apple'; // iPad, iPhone or iPod Touch
 53 	}
 54 	// Other WebKit Mobile Browsers
 55 	else if ((m = ua.match(/NokiaN[^\/]*|Android \d\.\d|webOS\/\d\.\d/))) {
 56 		o[MOBILE] = m[0].toLowerCase(); // Nokia N-series, Android, webOS, ex: NokiaN95
 57 	}
 58 }
 59 // NOT WebKit
 60 else {
 61 	// Presto
 62 	// ref: http://www.useragentstring.com/pages/useragentstring.php
 63 	if ((m = ua.match(/Presto\/([\d.]*)/)) && m[1]) {
 64 		o[core = 'presto'] = numberify(m[1]);
 65 		
 66 		// Opera
 67 		if ((m = ua.match(/Opera\/([\d.]*)/)) && m[1]) {
 68 			o[shell = 'opera'] = numberify(m[1]); // Opera detected, look for revision
 69 
 70 			if ((m = ua.match(/Opera\/.* Version\/([\d.]*)/)) && m[1]) {
 71 				o[shell] = numberify(m[1]);
 72 			}
 73 
 74 			// Opera Mini
 75 			if ((m = ua.match(/Opera Mini[^;]*/)) && m) {
 76 				o[MOBILE] = m[0].toLowerCase(); // ex: Opera Mini/2.0.4509/1316
 77 			}
 78 			// Opera Mobile
 79 			// ex: Opera/9.80 (Windows NT 6.1; Opera Mobi/49; U; en) Presto/2.4.18 Version/10.00
 80 			// issue: 由于 Opera Mobile 有 Version/ 字段，可能会与 Opera 混淆，同时对于 Opera Mobile 的版本号也比较混乱
 81 			else if ((m = ua.match(/Opera Mobi[^;]*/)) && m){
 82 				o[MOBILE] = m[0];
 83 			}
 84 		}
 85 		
 86 	// NOT WebKit or Presto
 87 	} else {
 88 		// MSIE
 89 		if ((m = ua.match(/MSIE\s([^;]*)/)) && m[1]) {
 90 			o[core = 'trident'] = 0.1; // Trident detected, look for revision
 91 			// 注意：
 92 			// o.shell = ie, 表示外壳是 ie
 93 			// 但 o.ie = 7, 并不代表外壳是 ie7, 还有可能是 ie8 的兼容模式
 94 			// 对于 ie8 的兼容模式，还要通过 documentMode 去判断。但此处不能让 o.ie = 8, 否则
 95 			// 很多脚本判断会失误。因为 ie8 的兼容模式表现行为和 ie7 相同，而不是和 ie8 相同
 96 			o[shell = 'ie'] = numberify(m[1]);
 97 
 98 			// Get the Trident's accurate version
 99 			if ((m = ua.match(/Trident\/([\d.]*)/)) && m[1]) {
100 				o[core] = numberify(m[1]);
101 			}
102 
103 		// NOT WebKit, Presto or IE
104 		} else {
105 			// Gecko
106 			if ((m = ua.match(/Gecko/))) {
107 				o[core = 'gecko'] = 0.1; // Gecko detected, look for revision
108 				if ((m = ua.match(/rv:([\d.]*)/)) && m[1]) {
109 					o[core] = numberify(m[1]);
110 				}
111 
112 				// Firefox
113 				if ((m = ua.match(/Firefox\/([\d.]*)/)) && m[1]) {
114 					o[shell = 'firefox'] = numberify(m[1]);
115 				}
116 			}
117 		}
118 	}
119 }
120 
121 o.core = core;
122 o.shell = shell;
123 o._numberify = numberify;
124 
125 });
