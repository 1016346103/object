664 		// 最后传进context的参数
665 		var args = [];
666 		// 存储所有uses，最后通过第一个参数传给context
667 		var runtime = host;
668 
669 		// host.__name__ 为自己的名称
670 		host.__name__ = options.name || pkg.name;
671 		// sys.modules
672 		if (host.__name__ === 'sys') host.modules = modules;
673 
674 		var done = function() {
675 			args.unshift(runtime);
676 			pkg.fn.apply(host, args);
677 			// 输出 __name__
678 			Object.keys(host).forEach(function(key) {
679 				if (typeof host[key] == 'function') {
680 					host[key].__name__ = key;
681 				}
682 			});
683 
684 			if (callback) callback();
685 		};
686 
687 		// 在没有uses的情况下直接返回即可。
688 		if (pkg.fn && pkg.uses.length === 0) {
689 			done();
690 			return;
691 		} else if (!pkg.fn && pkg.file) {
692 			self.loadScript(pkg.file, function() {
693 				loadNext(0);
694 			}, true);
695 			return;
696 		}
697 
698 		// 主递归函数
699 		function loadNext(i) {
700 
701 			var use = pkg.uses[i];
702 
703 			self.getModule(use.name, modules, function(useModule) {
704 				var names, root, member;
705 
706 				// 有别名
707 				if (use.as) {
708 					runtime[use.as] = useModule;
709 					if (args.indexOf(useModule) == -1) args.push(useModule);
710 
711 				// 是一个from import
712 				} else if (use.uses) {
713 					if (use.uses == '*') {
714 						for (member in useModule) {
715 							if (useModule.hasOwnProperty(member) && !member.match(/__.*__/g)) runtime[member] = useModule[member];
716 						}
717 					} else {
718 						use.uses.forEach(function(member) {
719 							if (member.as) runtime[member.as] = useModule[member.name];
720 							else runtime[member.name] = useModule[member.name];
721 						});
722 					}
723 
724 				// 正常的引入
725 				} else {
726 					names = use.name.split('.');
727 					root = runtime[names[0]] = modules[names[0]];
728 
729 					if (args.indexOf(root) == -1) args.push(root);
730 				}
731 
732 				if (i < pkg.uses.length - 1) {
733 					loadNext(i + 1);
734 				} else if (i == pkg.uses.length - 1) {
735 					done();
736 				}
737 
738 			});
739 
740 		};
741 
742 		loadNext(0);
743 
744 	};
745 
746 	/**
747 	 * 通过一个模块名，获得到相对应的模块对象并通过callback返回
748 	 *
749 	 * @param name pkg name
750 	 * @param modules 已引入的module对象列表，会传递给 execute 方法，可以通过sys.modules获取
751 	 * @param callback 模块获取到以后，通过callback的第一个参数传递回去
752 	 * @returns 最终引入的模块
753 	 */
754 	this.getModule = function(self, name, modules, callback) {
755 		var names = name.split('.');
756 
757 		/**
758 		 * @param i
759 		 * @param pname 上一个module的name
760 		 */
761 		function loadNext(i, pname) {
762 			var prefix = names.slice(0, i + 1).join('.');
763 			name = names[i];
764 
765 			var next = function() {
766 				if (pname) modules[pname][name] = modules[prefix];
767 
768 				if (i < names.length - 1) {
769 					loadNext(i + 1, prefix);
770 				} else if (i == names.length - 1) {
771 					callback(modules[prefix]);
772 				}
773 			};
774 
775 			// 使用缓存中的
776 			if (modules[prefix]) {
777 				next();
778 
779 			// lib 中有
780 			} else if (_lib[prefix]) {
781 				var pkg = _lib[prefix];
782 
783 				if (!modules[prefix]) {
784 					modules[prefix] = new Module(prefix);
785 				}
786 
787 				// 有可能是空的模块，是没有 fn 的
788 				if (pkg.fn) {
789 					self.executeModule(pkg, modules[prefix], modules, next);
790 
791 				// lib中有，但是是file，需要动态加载
792 				} else if (pkg.file) {
793 					// 文件加载完毕后，其中执行的 add 会自动把 _lib 中的对象替换掉，file 属性丢失，加入了 execute/name/uses 等属性
794 					// 使用缓存
795 					self.loadScript(pkg.file, function() {
796 						self.executeModule(pkg, modules[prefix], modules, next);
797 					}, true);
798 				} else {
799 					next();
800 				}
801 
802 			// lib中没有
803 			} else {
804 				throw new NoModuleError(prefix);
805 			}
806 
807 		};
808 
809 		loadNext(0);
810 	};
811 
812 	/**
813 	 * 处理传入的uses参数，输入格式：
814 	 * ['module1', {
815 	 * 	name: 'module2',
816 	 * 	uses: [
817 	 * 		{
818 	 * 			name: 'Member1',
819 	 * 			as: 'M'
820 	 * 		},
821 	 * 		'Member2'
822 	 * 	]
823 	 * }, {
824 	 * 	name: 'full.module.name',
825 	 * 	as: 'shortname'
826 	 * }, {
827 	 *	name: 'somemodule',
828 	 *	uses: '*'
829 	 * }];
830 	 *
831 	 * 输出格式：
832 	 * [{
833 	 * 	name: 'module1'
834 	 * }, {
835 	 *	name: 'module2',
836 	 *	uses: [
837 	 *		{
838 	 *			name: 'Member1',
839 	 *			as: 'M'
840 	 *		}, {
841 	 *			name: 'Member2'
842 	 *		}
843 	 *	]
844 	 * }, {
845 	 *	name: 'full.module.name',
846 	 *	as: 'shortname'
847 	 * }, {
848 	 *	name: 'somemodule',
849 	 *	uses: '*'
850 	 * }];
851 	 *
852 	 * 在getUses阶段不需要根据名称判断去重（比如自己use自己），因为并不能避免所有冲突，还有循环引用的问题（比如 core use dom, dom use core）
853 	 * 因此限定：globalUses 中声明的模块不允许有use
854 	 *
855 	 * @param uses 输入
856 	 * @param ignore 跳过ignore模块，用来避免自己调用自己
857 	 */
858 	this.getUses = function(self, uses, ignore) {
859 		if (typeof uses == 'string') {
860 
861 			uses = uses.split(/\s*,\s*/ig);
862 			for (var i = 0; i < uses.length; i++) {
863 				uses[i] = {name: uses[i]};
864 			}
865 
866 		} else {
867 
868 			for (i = 0, use; i < uses.length; i++) { // typeof uses == object
869 				use = uses[i];
870 				if (typeof use == 'string') {
871 					uses[i] = {name : use};
872 					continue;
873 				}
874 				if (use.uses) {
875 					// uses/as 互斥
876 					use.as = null;
877 					if (use.uses == '*') continue; // 如果是 * 则保持字符串，不做处理
878 					if (typeof use.uses == 'string') {
879 						use.uses = use.uses.split(/\s*,\s*/ig);
880 					}
881 					for (var j = 0, subuse; j < use.uses.length; j++) {
882 						subuse = use.uses[j];
883 						if (typeof subuse == 'string') {
884 							use.uses[j] = {
885 								name: subuse
886 							};
887 						}
888 					}
889 					continue;
890 				}
891 			}
892 
893 		}
894 
895 		// 过滤自己调用自己
896 		uses = uses.filter(function(use) {
897 			return use.name != ignore;
898 		});
899 
900 		return uses;
901 	};
902 
903 	/**
904 	 * 传入context，context的参数会包含use进来的module
905 	 * 创造一个context，内部通过 this.xxx 设置的成员都会在这个 context 下。
906 	 * @param name 名称
907 	 * @param uses 用逗号分隔开的模块名称列表
908 	 * @param context 这个function会在调用module时调用，并将module通过参数传入context，第一个参数为runtime，后面的参数为每个module的不重复引用，顺序排列
909 	 */
910 	this.add = function(self, name, uses, context) {
911 
912 		// 不允许重复添加。
913 		if (_lib[name] && _lib[name].fn) return null;
914 
915 		// uses 参数是可选的
916 		if (typeof uses == 'function') {
917 			context = uses;
918 			uses = [];
919 		}
920 
921 		var globalUses = self.globalUses? self.getUses(self.globalUses, name) : [];
922 		// globalUses放在数组前面还是后面还有大讲究：
923 		// 放在前面，首先引入的core模块无法通过sys.modules找到其他后加载的模块；
924 		// 放在后面，则可以找到。
925 		uses = self.getUses(uses, name).concat(globalUses);
926 		//uses = globalUses.concat(self.getUses(uses));
927 
928 		// 建立前缀占位模块
929 		self.makePrefixPackage(name);
930 
931 		// lib中存储的是function
932 		// 注意别给覆盖了，有可能是有 file 成员的
933 		var pkg = _lib[name];
934 		if (!pkg) pkg = _lib[name] = {};
935 		pkg.name = name;
936 		pkg.uses = uses;
937 		pkg.fn = context;
938 
939 		return pkg;
940 	};
941 
942 	/**
943 	 * use
944 	 * uses参数书写有一个技巧，写在后面的module在运行时可以通过runtime参数得到之前已经引入module的引用
945 	 * 这对于利用引入对象做处理的模块来说是一个很重要的特性
946 	 * @param uses 用逗号分隔开的模块名称列表
947 	 * @param context uses加载后调用，将module通过参数传入context，第一个参数为runtime，后面的参数为每个module的不重复引用，顺序排列
948 	 */
949 	this.use = function(self, uses, context) {
950 		self.loadLib();
951 
952 		var name = '__anonymous_' + self.anonymousModuleCount + '__';
953 		self.anonymousModuleCount++;
954 		var module = self.add(name, uses, context);
955 
956 		// 第二个{}参数会被所有相关module通过第一个 runtime 参数获取到，实现module获取调用者的信息
957 		self.executeModule(module, {}, {}, null, {name: '__main__'});
958 	};
959 
960 	/**
961 	 * execute
962 	 * @param name 执行的入口模块名称
963 	 * @param options 传入参数
964 	 */ 
965 	this.execute = function(self, name, options) {
966 		self.loadLib();
967 
968 		var module = _lib[name];
969 		if (!module) throw new NoModuleError(name);
970 
971 		self.executeModule(module, {}, {}, null, {name: '__main__'});
972 	};
973 
974 });
