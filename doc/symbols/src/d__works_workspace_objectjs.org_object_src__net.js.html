  5 object.add('net', 'dom, events', /**@lends net*/ function(exports, dom, events) {

  7 var ajaxProxies = this.ajaxProxies = {};

  9 // 执行一个可跨域的ajax请求
 10 // 跨域host必须有ajaxproxy.htm
 11 // callback唯一参数返回 XMLHttpRequest 对象实例
 12 this.ajaxRequest = function(url, callback) {
 13 	var tmpA = document.createElement('a');
 14 	tmpA.href = url;
 15 	var hostname = tmpA.hostname;
 16 
 17 	if (hostname && (hostname != location.hostname)) {
 18 		var xhr = null;
 19 		if (ajaxProxies[hostname]) callback(ajaxProxies[hostname].getTransport());
 20 		else {
 21 			var iframe = document.createElement('iframe');
 22 			iframe.style.display = 'none';
 23 			dom.ready(function() {
 24 				document.body.insertBefore(iframe, document.body.firstChild);
 25 				iframe.src = 'http://' + hostname + '/ajaxproxy.htm';
 26 				if (iframe.attachEvent) {
 27 					iframe.attachEvent('onload', function () {
 28 						callback(iframe.contentWindow.getTransport());
 29 						ajaxProxies[hostname] = iframe.contentWindow;
 30 					});
 31 				} else {
 32 					iframe.onload = function () {
 33 						callback(iframe.contentWindow.getTransport());
 34 						ajaxProxies[hostname] = iframe.contentWindow;
 35 					};
 36 				}
 37 			});
 38 		}
 39 	} else {
 40 		if (window.ActiveXObject) {
 41 			try {
 42 				callback(new ActiveXObject('Msxml2.XMLHTTP'));
 43 			} catch(e) {
 44 				callback(new ActiveXObject('Microsoft.XMLHTTP'));
 45 			}
 46 		} else callback(new XMLHttpRequest());
 47 	}
 48 };

 50 this.ping = function(url) {
 51 	var n = "_net_ping_"+ (new Date()).getTime();
 52 	var c = window[n] = new Image(); // 把new Image()赋给一个全局变量长期持有
 53 	c.onload = (c.onerror=function(){window[n] = null;});
 54 	c.src = url;
 55 	c = null; // 释放局部变量c
 56 };

 58 /**
 59  * 发送Ajax请求的类
 60  * 使用时需要实例化一个Request对象,然后手动调用该对象的send方法完成发送(与base中的xmlhttp不同)
 61  * 
 62  * @param {object} options
 63  * @param {string} options.url 要请求的url
 64  * @param {string} options.method get/post
 65  * @param {function} options.onsuccess 请求成功后的回调,参数是封装过的ajax对象
 66  * param {function} options.onerror 请求失败后的回调
 67  */
 68 this.Request = new Class(function() {
 69 
 70 	Class.mixin(this, events.Events);
 71 
 72 	this.initialize = function(self, options) {
 73 		self.url = options.url || '';
 74 		self.method = options.method || 'get';
 75 		self.headers = {};
 76 		self._xhr = null;
 77 
 78 		self.onSuccess = options.onSuccess;
 79 		self.onsuccess = options.onsuccess;
 80 		self.onerror = options.onerror;
 81 		self.oncomplete = options.oncomplete;
 82 	};
 83 
 84 	this.send = function(self, params) {
 85 		exports.ajaxRequest(self.url, function(xhr) {
 86 			self._xhr = xhr;
 87 			var eventData = {request: xhr};
 88 
 89 			xhr.onreadystatechange = function() {
 90 				var xhr = self._xhr;
 91 
 92 				if (xhr.readyState === 4) {
 93 
 94 					// IE6 dont's support getResponseHeader method
 95 					//if (xhr.getResponseHeader('Content-Type') == 'text/json') {
 96 						//xhr.responseJSON = JSON.parse(xhr.responseText)
 97 					//}
 98 
 99 					// Compatible
100 					eventData.responseText = xhr.responseText;
101 					eventData.responseXML = xhr.responseXML;
102 					//eventData.responseJSON = xhr.responseJSON;
103 
104 					if (xhr.status === undefined || xhr.status === 0 || (xhr.status >= 200 && xhr.status < 300)) {
105 						self.fireEvent('success', eventData);
106 						if (self.onSuccess) self.onSuccess(eventData);
107 					} else {
108 						self.fireEvent('error', eventData);
109 					}
110 					self.fireEvent('complete', eventData);
111 				}
112 			};
113 			var xhr = self._xhr;
114 			var url = self.url;
115 
116 			// 处理params
117 			if (params && self.method == 'get') {
118 				url += (url.indexOf('?') != -1 ? '&' : '?') + params;
119 				params = null;
120 			}
121 
122 			// open
123 			xhr.open(self.method, url, true);
124 
125 			xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
126 
127 			// headers
128 			for (var name in self.headers) {
129 				xhr.setRequestHeader(name, self.headers[name]);
130 			}
131 
132 			self._xhr.send(params);
133 		});
134 	};
135 
136 	this.setHeader = function(self, name, value) {
137 		self.headers[name] = value;
138 	};
139 
140 });
141 
142 });
