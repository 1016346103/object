1 /**
  2  * @namespace
  3  * @name options
  4  */
  5 object.add('options', /**@lends options*/ function(exports) {
  6 
  7 // 仿照 mootools 的overloadSetter，返回一个 key/value 这种形式的function参数的包装，使其支持{key1: value1, key2: value2} 这种形式
  8 var enumerables = true;
  9 for (var i in {toString: 1}) enumerables = null;
 10 if (enumerables) enumerables = ['hasOwnProperty', 'valueOf', 'isPrototypeOf', 'propertyIsEnumerable', 'toLocaleString', 'toString', 'constructor'];
 11 // func有可能是个method，需要支持传递self参数
 12 this.overloadsetter = function(func) {
 13 	return function() {
 14 		var a = arguments[func.length - 2] || null;
 15 		var b = arguments[func.length - 1];
 16 		var passArgs = args = Array.prototype.slice.call(arguments, 0, func.length - 2);
 17 
 18 		if (a === null) return this;
 19 		if (typeof a != 'string') {
 20 			for (var k in a) {
 21 				args = passArgs.slice(0); // 复制，否则循环多次参数就越来越多了
 22 				args.push(k);
 23 				args.push(a[k]);
 24 				func.apply(this, args);
 25 			}
 26 			if (enumerables) {
 27 				for (var i = enumerables.length; i > 0; i--) {
 28 					k = enumerables[i];
 29 					if (a.hasOwnProperty(k)) func.call(this, k, a[k]);
 30 				}
 31 			}
 32 		} else {
 33 			args.push(a);
 34 			args.push(b);
 35 			func.apply(this, args);
 36 		}
 37 		return this;
 38 	};
 39 };

 41 /**
 42  * 这个类辅助这种参数传递方式的实现：
 43  * callFunc({
 44  *	param1: someValue1,
 45  *	param2: someValue2
 46  * })
 47  * 在声明函数时，通过：
 48  * var opts = new ns.Arguments(opts, {
 49  *	param1: 1,
 50  *	param2: 2
 51  * });
 52  * 来设定默认值，没有设置过默认值的成员不会输出
 53  * @class
 54  */
 55 this.Arguments = new Class(function() {
 56 
 57 	/**
 58 	 * @param defaults 默认值列表
 59 	 * @param opts 参数列表
 60 	 */
 61 	this.initialize = function(self, defaults, opts) {
 62 		if (opts == undefined) opts = {};
 63 
 64 		var output = {};
 65 		for (var key in defaults) {
 66 			output[key] = (opts[key] != undefined? opts[key] : defaults[key]);
 67 		}
 68 		return output;
 69 	};
 70 
 71 });

 73 /**
 74  * 参数
 75  * @class
 76  */
 77 this.Options = new Class({
 78 
 79 	/**
 80 	 * 提供一个实现了 makeOption 接口的“提供者”参数，这样，在 setOption 时会自动根据name获取value，不用手工调用
 81 	 */
 82 	initialize: function(self, provider) {
 83 		if (provider) self._provider = provider;
 84 		self._options = {};
 85 	},
 86 
 87 	setOptions: function(self, options, host) {
 88 		if (!host) host = self._options;
 89 
 90 		for (var i in options) {
 91 			if (host[i] !== undefined) host[i] = options[i];
 92 		}
 93 	},
 94 
 95 	setOption: function(self, name, type, value) {
 96 		if (value !== undefined) {
 97 			self._options[name] = value;
 98 		} else {
 99 			value = self._provider.makeOption(name, type);
100 			if (value === null) return;
101 			else self._options[name] = value;
102 		}
103 	},
104 
105 	getOptions: function(self) {
106 		return self._options;
107 	}
108 
109 });
110 
111 });
