 10 IEEvent.prototype.stopPropagation = function() {
 11 	this.cancelBubble = true;

 14 IEEvent.prototype.preventDefault = function() {
 15 	this.returnValue = false;

 18 IEEvent.prototype.getPreventDefault = function() {
 19 	// 自定义事件是没有 returnValue 值的，如果设置默认为true，则会导致非自定义的事件后面再设置false失效，出现无法preventDefault()的问题
 20 	// 不能设置默认值，就只能严格限制returnValue === false才算preventDefaulted
 21 	return this.returnValue === false;
 22 };

 24 IEEvent.prototype.stop = function() {
 25 	this.stopPropagation();
 26 	this.preventDefault();
 27 };

 29 /**
 30  * decorator
 31  * 使得相应方法在调用时fire出同名事件，并支持preventDefault
 32  * fireevent 或 fireevent(eventName)
 33  * fireevent 默认eventName通过__name__获得
 34  */
 35 this.fireevent = function(arg1) {
 36 	var name, func, eventDataNames;
 37 
 38 	// 千万别给这个function起名字，否则fire出来的事件都叫一个名字
 39 	var firer = function(self) {
 40 		var nativeName = Function.__get_name__(arguments.callee) || arguments.callee.__name__;
 41 		if (!name) name = nativeName;
 42 
 43 		// 根据eventDataNames生成eventData，每一个参数对应一个eventData
 44 		var eventData = {};
 45 		if (eventDataNames) {
 46 			for (var i = 0; i < eventDataNames.length; i++) {
 47 				// 名字对应方法的参数，从第2个参数开始，因为第一个是self
 48 				eventData[eventDataNames[i]] = arguments[i + 1];
 49 			}
 50 		}
 51 
 52 		var event = self.fireEvent(name, eventData, self);
 53 
 54 		// 执行 xxx_createEvent 方法，可用于定制event
 55 		var createEventMethod = self[nativeName + '_createEvent'];
 56 		if (createEventMethod) {
 57 			var args = Array.prototype.slice.call(arguments, 1);
 58 			args.unshift(event);
 59 			createEventMethod.apply(self, args);
 60 		}
 61 
 62 		// Webkit 使用 defaultPrevented
 63 		// Gecko 使用 getPreventDefault()
 64 		// IE 用 returnValue 模拟了 getPreventDefault
 65 		var preventDefaulted = event.getPreventDefault? event.getPreventDefault() : event.defaultPrevented;
 66 		if (!preventDefaulted) func.apply(this, arguments);
 67 		return event;
 68 	};
 69 
 70 	if (typeof arg1 == 'function') {
 71 		func = arg1;
 72 		return firer;
 73 
 74 	// 自定义了事件名称，返回一个decorator
 75 	} else {
 76 		if (Array.isArray(arguments[0])) {
 77 			eventDataNames = arguments[0];
 78 		} else {
 79 			name = arg1;
 80 			if (arguments[1]) eventDataNames = arguments[1];
 81 		}
 82 		return function(_func) {
 83 			func = _func;
 84 			return firer;
 85 		};
 86 	}
 87 
 88 };

 90 /**
 91  * 将IE中的window.event包装一下

 93 this.wrapEvent = function(e) {
 94 	// 之前手贱在这里写了个 e.returnValue = true
 95 	// 于是所有的事件都无法阻止执行了
 96 	// IE可能只认第一次赋值，因为后面还是有重新把returnValue设置成false的
 97 
 98 	e.target = e.srcElement;
 99 	e.stopPropagation = IEEvent.prototype.stopPropagation;
100 	e.preventDefault = IEEvent.prototype.preventDefault;
101 	e.getPreventDefault = IEEvent.prototype.getPreventDefault;
102 	e.stop = IEEvent.prototype.stop;
103 
104 	return e;
105 };

107 // 事件
108 this.Events = new Class(/**@lends object.Event*/ function() {
109 
110 	this.__addEvent = function(self, type, func, cap) {
111 		var propertyName = '_event_' + type;
112 		if (self[propertyName] === undefined) {
113 			self[propertyName] = 0;
114 		}
115 		self.attachEvent('onpropertychange', function(event) {
116 			if (event.propertyName == propertyName) {
117 				func();
118 			}
119 		});
120 	};
121 
122 	this.initialize = function(self) {
123 		if (!self.addEventListener) {
124 			self._eventListeners = {};
125 		}
126 		// 自定义事件，用一个隐含div用来触发事件
127 		if (!self.addEventListener && !self.attachEvent) {
128 			self.__boss = document.createElement('div');
129 		}
130 	};
131 
132 	/**
133 	 * 添加事件
134 	 * @param self
135 	 * @param type 事件名
136 	 * @param func 事件回调
137 	 * @param cap 冒泡
138 	 */
139 	this.addEvent = document.addEventListener? function(self, type, func, cap) {
140 		var boss = self.__boss || self;
141 
142 		if (cap === null) cap = false;
143 
144 		// 非IE不支持mouseleave/mouseenter事件
145 		// 在老base中大量使用了这个事件，支持一下
146 		if (!ua.ua.ie && type == 'mouseleave') {
147 			var ismouseleave = function(event, element) {
148 				var p = event.relatedTarget;
149 				while ( p && p != element ) try { p = p.parentNode; } catch(error) { p = element; }
150 				return p !== element;
151 			};
152 			var innerFunc = func;
153 			func = function(event) {
154 				var p = event.relatedTarget;
155 				while (p && p != self) try {
156 					p = p.parentNode;
157 				} catch (e) {
158 					p = self;
159 				}
160 				if (p !== self && innerFunc) innerFunc.call(self, event);
161 			};
162 			func.innerFunc = innerFunc;
163 			type = 'mouseout';
164 		}
165 
166 		boss.addEventListener(type, func, cap);
167 	} : function(self, type, func) {
168 		var boss = self.__boss || self;
169 
170 		// 存储此元素的事件
171 		if (!self._eventListeners[type]) {
172 			self._eventListeners[type] = [];
173 		}
174 		var funcs = self._eventListeners[type];
175 
176 		// 不允许两次添加同一事件
177 		if (funcs.some(function(f) {
178 			return f.innerFunc === func;
179 		})) return;
180 
181 		// 为IE做事件包装，使回调的func的this指针指向元素本身，并支持preventDefault等
182 		// 包装Func，会被attachEvent
183 		// 包装Func存储被包装的func，detach的时候，参数是innerFunc，需要通过innerFunc找到wrapperFunc进行detach
184 		var wrapperFunc = function(eventData) {
185 			var e = arguments.length > 1? eventData : exports.wrapEvent(window.event);
186 			func.call(self, e);
187 		};
188 		wrapperFunc.innerFunc = func;
189 
190 		funcs.push(wrapperFunc);
191 
192 		boss.attachEvent('on' + type, wrapperFunc);
193 	};
194 
195 	/**
196 	 * 移除事件
197 	 * @param self
198 	 * @param type 事件名
199 	 * @param func 事件回调
200 	 * @param cap 冒泡
201 	 */
202 	this.removeEvent = document.removeEventListener? function(self, type, func, cap) {
203 		var boss = self.__boss || self;
204 
205 		boss.removeEventListener(type, func, cap);
206 	} : function(self, type, func, cap) {
207 		var boss = self.__boss || self;
208 
209 		if (!self._eventListeners) self._eventListeners = {};
210 		var funcs = self._eventListeners[type];
211 		if (!funcs) return;
212 
213 		// func 是 innerFunc，需要找到 wrapperFunc
214 		for (var i = 0, wrapperFunc; i < funcs.length; i++) {
215 			wrapperFunc = funcs[i];
216 			if (wrapperFunc === func || wrapperFunc.innerFunc === func) {
217 				funcs.splice(i, 1); // 将这个function删除
218 				break;
219 			}
220 		}
221 		// 如果没有找到func，虽然此次remove无效，但是根据标准，不应该报错。
222 		if (wrapperFunc) boss.detachEvent('on' + type, wrapperFunc);
223 	};
224 
225 	/**
226 	 * 触发事件
227 	 * @param self
228 	 * @param type 事件名
229 	 * @param eventData 扩展到event对象上的数据
230 	 */
231 	this.fireEvent = document.dispatchEvent? function(self, type, eventData) {
232 		var boss = self.__boss || self;
233 
234 		var triggerName = 'on' + type.toLowerCase();
235 		var event = document.createEvent('Event');
236 		event.initEvent(type, false, true);
237 		extend(event, eventData);
238 
239 		if (self[triggerName]) {
240 			var returnValue = self[triggerName].call(self, event);
241 			if (returnValue === false) event.preventDefault();
242 		}
243 
244 		boss.dispatchEvent(event);
245 		return event;
246 	} : function(self, type, eventData) {
247 		if (!eventData) eventData = {};
248 		var triggerName = 'on' + type.toLowerCase();
249 		var event = exports.wrapEvent(eventData);
250 
251 		if (self[triggerName]) {
252 			var returnValue = self[triggerName].call(self, event);
253 			if (returnValue === false) event.preventDefault();
254 		}
255 
256 		if (!self._eventListeners[type]) return event;
257 		var funcs = self._eventListeners[type];
258 		for (var i = 0, j = funcs.length; i < j; i++) {
259 			if (funcs[i]) {
260 				funcs[i].call(self, event, true);
261 			}
262 		}
263 		return event;
264 	};
265 });
