 10 IEEvent.prototype.stopPropagation = function() {
 11 	this.cancelBubble = true;

 14 IEEvent.prototype.preventDefault = function() {

 18 IEEvent.prototype.getPreventDefault = function() {
 19 	// 自定义事件是没有 returnValue 值的，如果设置默认为true，则会导致非自定义的事件后面再设置false失效，出现无法preventDefault()的问题
 20 	// 不能设置默认值，就只能严格限制returnValue === false才算preventDefaulted
 21 	return this.returnValue === false;
 22 };

 24 IEEvent.prototype.stop = function() {
 25 	this.stopPropagation();
 26 	this.preventDefault();
 27 };

 29 /**
 30  * decorator
 31  * 使得相应方法在调用时fire出同名事件，并支持preventDefault
 32  * fireevent 或 fireevent(eventName)
 33  * fireevent 默认eventName通过__name__获得
 34  */
 35 this.fireevent = function(arg1) {
 36 	var name, func, eventDataNames;
 37 
 38 	// 千万别给这个function起名字，否则fire出来的事件都叫一个名字
 39 	var firer = function(self) {
 40 		// 获取function原生name似乎没什么用
 41 		// var nativeName = Function.__get_name__(arguments.callee) || arguments.callee.__name__;
 42 		var nativeName = arguments.callee.__name__;
 43 		if (!name) name = nativeName;
 44 
 45 		// 根据eventDataNames生成eventData，每一个参数对应一个eventData
 46 		var eventData = {};
 47 		// 保存func被调用时的所有参数（除了self）
 48 		var args = Array.prototype.slice.call(arguments, 1);
 49 		if (eventDataNames) {
 50 			for (var i = 0; i < eventDataNames.length; i++) {
 51 				// 名字对应方法的参数，从第2个参数开始，因为第一个是self
 52 				eventData[eventDataNames[i]] = arguments[i + 1];
 53 			}
 54 		}
 55 		// 默认有一个_args的data，
 56 		eventData._args = args;
 57 
 58 		var event = self.fireEvent(name, eventData, self);
 59 
 60 		// 执行 xxx_createEvent 方法，可用于定制event
 61 		var createEventMethod = self[nativeName + '_createEvent'];
 62 		if (createEventMethod) {
 63 			args.unshift(event);
 64 			createEventMethod.apply(self, args);
 65 		}
 66 
 67 		// Webkit 使用 defaultPrevented
 68 		// Gecko 使用 getPreventDefault()
 69 		// IE 用 returnValue 模拟了 getPreventDefault
 70 		var preventDefaulted = event.getPreventDefault? event.getPreventDefault() : event.defaultPrevented;
 71 		if (!preventDefaulted) return func.apply(this, arguments);
 72 	};
 73 
 74 	if (typeof arg1 == 'function') {
 75 		func = arg1;
 76 		return firer;
 77 
 78 	// 自定义了事件名称，返回一个decorator
 79 	} else {
 80 		if (Array.isArray(arguments[0])) {
 81 			eventDataNames = arguments[0];
 82 		} else {
 83 			name = arg1;
 84 			if (arguments[1]) eventDataNames = arguments[1];
 85 		}
 86 		return function(_func) {
 87 			func = _func;
 88 			return firer;
 89 		};
 90 	}
 91 
 92 };

 94 /**
 95  * 将IE中的window.event包装一下
 96  */
 97 this.wrapEvent = function(e) {
 98 	// 之前手贱在这里写了个 e.returnValue = true
 99 	// 于是所有的事件都无法阻止执行了
100 	// IE可能只认第一次赋值，因为后面还是有重新把returnValue设置成false的
101 
102 	e.target = e.srcElement;
103 	e.stopPropagation = IEEvent.prototype.stopPropagation;
104 	e.preventDefault = IEEvent.prototype.preventDefault;
105 	e.getPreventDefault = IEEvent.prototype.getPreventDefault;
106 	e.stop = IEEvent.prototype.stop;
107 
108 	return e;

112 this.Events = new Class(/**@lends events.Event*/ function() {
113 
114 	// 在标准浏览器中使用的是系统事件系统，无法保证nativeEvents在事件最后执行。
115 	// 需在每次addEvent时，都将nativeEvents的事件删除再添加，保证在事件队列最后，最后才执行。
116 	function moveNativeEventsToTail(self, type) {
117 		var boss = self.__boss || self;
118 		if (self.__nativeEvents && self.__nativeEvents[type]) {
119 			// 删除之前加入的
120 			boss.removeEventListener(type, self.__nativeEvents[type].run, false);
121 			// 重新添加到最后
122 			boss.addEventListener(type, self.__nativeEvents[type].run, false);
123 		}
124 	};
125 
126 	function handle(self, type) {
127 		var boss = self.__boss || self;
128 		boss.attachEvent('on' + type, function(eventData) {
129 			var event = arguments.length > 1? eventData : exports.wrapEvent(window.event);
130 			var funcs = self.__eventListeners? self.__eventListeners[type] : null;
131 			if (funcs) {
132 				funcs.forEach(function(func) {
133 					try {
134 						func.call(self, event);
135 					} catch(e) {
136 					}
137 				});
138 			}
139 			var natives = self.__nativeEvents? self.__nativeEvents[type] : null;
140 			if (natives) {
141 				natives.forEach(function(func) {
142 					func.call(self, event);
143 				});
144 			}
145 		});
146 	}
147 
148 	this.initialize = function(self) {
149 		if (!self.addEventListener) {
150 			// 在一些情况下，你不知道传进来的self对象的情况，不要轻易的将其身上的__eventListeners清除掉
151 			if (!self.__eventListeners) self.__eventListeners = {};
152 			if (!self.__nativeEvents) self.__nativeEvents = {};
153 		}
154 		// 自定义事件，用一个隐含div用来触发事件
155 		if (!self.addEventListener && !self.attachEvent) {
156 			self.__boss = document.createElement('div');
157 		}
158 	};
159 
160 	/**
161 	 * 添加事件
162 	 * @param self
163 	 * @param type 事件名
164 	 * @param func 事件回调
165 	 * @param cap 冒泡
166 	 */
167 	this.addEvent = document.addEventListener? function(self, type, func, cap) {
168 		var boss = self.__boss || self;
169 
170 		if (cap === null) cap = false;
171 
172 		// 非IE不支持mouseleave/mouseenter事件
173 		// 在老base中大量使用了这个事件，支持一下
174 		if (!ua.ua.ie && type == 'mouseleave') {
175 			var ismouseleave = function(event, element) {
176 				var p = event.relatedTarget;
177 				while ( p && p != element ) try { p = p.parentNode; } catch(error) { p = element; }
178 				return p !== element;
179 			};
180 			var innerFunc = func;
181 			func = function(event) {
182 				var p = event.relatedTarget;
183 				while (p && p != self) try {
184 					p = p.parentNode;
185 				} catch (e) {
186 					p = self;
187 				}
188 				if (p !== self && innerFunc) innerFunc.call(self, event);
189 			};
190 			func.innerFunc = innerFunc;
191 			type = 'mouseout';
192 		}
193 
194 		boss.addEventListener(type, func, cap);
195 		moveNativeEventsToTail(self, type);
196 
197 	} : function(self, type, func) {
198 		var boss = self.__boss || self;
199 
200 		// 存储此元素的事件
201 		var funcs;
202 		if (!self.__eventListeners) self.__eventListeners = {};
203 		if (!self.__eventListeners[type]) {
204 			funcs = [];
205 			self.__eventListeners[type] = funcs;
206 			if (!self.__nativeEvents || !self.__nativeEvents[type]) {
207 				handle(self, type);
208 			}
209 		} else {
210 			funcs = self.__eventListeners[type];
211 		}
212 
213 		// 不允许两次添加同一事件
214 		if (funcs.some(function(f) {
215 			return f === func;
216 		})) return;
217 
218 		funcs.push(func);
219 
220 	};
221 
222 	this.addNativeEvent = document.addEventListener? function(self, type, func) {
223 		var boss = self.__boss || self;
224 		var natives;
225 		if (!self.__nativeEvents) self.__nativeEvents = {};
226 		if (!self.__nativeEvents[type]) {
227 			natives = [];
228 			self.__nativeEvents[type] = natives;
229 			self.__nativeEvents[type].run = function(event) {
230 				natives.forEach(function(func) {
231 					func.call(self, event);
232 				});
233 			};
234 			moveNativeEventsToTail(self, type);
235 		} else {
236 			natives = self.__nativeEvents[type];
237 		}
238 		natives.push(func);
239 
240 	} : function(self, type, func) {
241 		var boss = self.__boss || self;
242 		var natives;
243 		if (!self.__nativeEvents) self.__nativeEvents = {};
244 		if (!self.__nativeEvents[type]) {
245 			natives = [];
246 			self.__nativeEvents[type] = natives;
247 			if (!self.__nativeEvents || !self.__eventListeners[type]) {
248 				handle(self, type);
249 			}
250 		} else {
251 			natives = self.__nativeEvents[type];
252 		}
253

256 			return f === func;

262 	/**
263 	 * 移除事件
264 	 * @param self
265 	 * @param type 事件名
266 	 * @param func 事件回调
267 	 * @param cap 冒泡
268 	 */
269 	this.removeEvent = document.removeEventListener? function(self, type, func, cap) {
270 		var boss = self.__boss || self;
271 
272 		boss.removeEventListener(type, func, cap);
273 	} : function(self, type, func, cap) {
274 		var boss = self.__boss || self;
275 
276 		if (!self.__eventListeners) self.__eventListeners = {};
277 		var funcs = self.__eventListeners[type];
278 		if (!funcs) return;
279 
280 		for (var i = 0; i < funcs.length; i++) {
281 			if (funcs[i] === func) {
282 				funcs.splice(i, 1); // 将这个function删除
283 				break;
284 			}
285 		}
286 	};

288 	/**
289 	 * 触发事件
290 	 * @param self
291 	 * @param type 事件名
292 	 * @param eventData 扩展到event对象上的数据
293 	 */
294 	this.fireEvent = document.dispatchEvent? function(self, type, eventData) {
295 		var boss = self.__boss || self;
296 
297 		var triggerName = 'on' + type.toLowerCase();
298 		var event = document.createEvent('Event');
299 		event.initEvent(type, false, true);
300 		object.extend(event, eventData);
301 
302 		if (self[triggerName]) {
303 			var returnValue = self[triggerName].call(self, event);
304 			if (returnValue === false) event.preventDefault();
305 		}
306 
307 		boss.dispatchEvent(event);
308 		return event;
309 	} : function(self, type, eventData) {
310 		if (!eventData) eventData = {};
311 		var triggerName = 'on' + type.toLowerCase();
312 		var event = exports.wrapEvent(eventData);
313 
314 		if (self[triggerName]) {
315 			var returnValue = self[triggerName].call(self, event);
316 			if (returnValue === false) event.preventDefault();
317 		}
318 
319 		if (!self.__eventListeners[type]) return event;
320 		var funcs = self.__eventListeners[type];
321 		for (var i = 0, j = funcs.length; i < j; i++) {
322 			if (funcs[i]) {
323 					try {
324 						funcs[i].call(self, event, true);
325 					} catch(e) {
326 					}
327 			}
328 		}
329 
330 		var natives = self.__nativeEvents[type];
331 		if (natives) {
332 			natives.forEach(function(func) {
333 				func.call(self, event);
334 			});
335 		}
336 
337 		return event;
338 	};
