 14 	Object.keys(dom.Element).forEach(function(name) {

 23 					arg = arguments[i];
 24 					args.push((arg && arg._node)? arg._node : arg);
 25 				}
 26 				return dom.Element[name].apply(dom.Element, args);
 27 			};
 28 		}

 85 this.ComponentClass = function(cls, name, base, members) {
 86 	var subEvents = {};
 87 	if (base && base._subEvents) {
 88 		Object.keys(base._subEvents).forEach(function(name) {
 89 			subEvents[name] = base._subEvents[name].slice(0); // 复制数组
 90 		});
 91 	}
 92 
 93 	var events = [];
 94 	if (base && base._events) {
 95 		events = base._events.slice(0);
 96 	}
 97 
 98 	var subs = {};
 99 	if (base && base._subs) {
100 		Object.keys(base._subs).forEach(function(name) {
101 			subs[name] = base._subs[name];
102 		});
103 	}
104 
105 	var defaultOptions = {};
106 	if (base && base._defaultOptions) {
107 		Object.keys(base._defaultOptions).forEach(function(name) {
108 			defaultOptions[name] = base._defaultOptions[name];
109 		});
110 	}
111 
112 	cls.__mixin__({
113 		_defaultOptions : defaultOptions,
114 		_subs : subs,
115 		_subEvents: subEvents,
116 		_events : events
117 	});
118 
119 	Object.keys(members).forEach(function(name) {
120 		var member = members[name];
121 		if (member.__class__ === property) {
122 			if (member.isComponent) {
123 				subs[name] = {
124 					selector: member.selector,
125 					type: member.type || exports.Component,
126 					single: member.single,
127 					nodeMap: {}, // 相应node的uid对应component，用于在需要通过node找到component时使用
128 					rendered: [] // 后来被加入的，而不是首次通过selector选择的node的引用
129 				};
130 			} else {
131 				cls._defaultOptions[name] = member.defaultValue;
132 			}
133 		} else if (typeof member == 'function') {
134 			var match = name.match(/^(_?[a-zA-Z]+)_([a-zA-Z]+)$/);
135 			if (match) {
136 				var component = match[1];
137 				var eventName = match[2];
138 				if (!subEvents[component]) subEvents[component] = [];
139 				if (subEvents[component].indexOf(eventName) === -1) subEvents[component].push(eventName);
140 
141 			} else if (name.slice(0, 1) == '_' && name.slice(0, 2) != '__' && name != '_set') { // _xxx but not __xxx
142 				eventName = name.slice(1);
143 				if (events.indexOf(eventName === -1)) events.push(eventName);
144 				cls.__mixin__(eventName, function(self) {
145 					self.fireEvent(eventName);
146 				});
147 			}
148 		}
149 	});
150 };

152 /**
153  * 为一个Component定义一个sub components引用
154  * 用法：
155  * MyComponent = new Class(ui.Component, {
156  *	refname: ui.define('css selector', ui.menu.Menu)
157  * });
158  * 这样MyComponent实例的refname属性极为相对应selector获取到的节点引用
159  * @param selector 选择器
160  * @param type 构造类
161  * @param single 是否是单独的引用
162  */
163 this.define = function(selector, type, single) {
164 	var prop = property(function(self) {
165 		return self[prop.__name__];
166 	});
167 	prop.isComponent = true;
168 	prop.selector = selector;
169 	prop.type = type;
170 	prop.single = single;
171 	return prop;
172 };

181 /**
182  * 声明一个option
183  * 用法：
184  * MyComponent = new Class(ui.Component, {
185  *	myConfig: ui.option(1)
186  * });
187  * 这样MyComponent实例的myConfig属性值即为默认值1，可通过 set 方法修改
188  */
189 this.option = function(defaultValue) {
190 	var prop;
191 	function fget(self) {
192 		return self.getOption(prop.__name__);
193 	}
194 	function fset(self, value) {
195 		return self.setOption(prop.__name__, value);
196 	}
197 	prop = property(fget, fset);
198 	prop.defaultValue = defaultValue;
199 	return prop;
200 };

202 /**
203  * UI模块基类，所有UI组件的基本类
204  * @class
205  * @name ui.Component
206  */
207 this.Component = new Class(/**@lends ui.Component*/ function() {
208 
209 	this.__metaclass__ = exports.ComponentClass;
210 
211 	var getConstructor = function(type) {
212 		if (type === 'number') return Number;
213 		else if (type === 'string') return String;
214 		else if (type === 'boolean') return Boolean;
215 	};
216 
217 	Class.mixin(this, Element);
218 
219 	this.initialize = function(self, node, options) {
220 		if (!options) options = {};
221 		self.__initSubOptions(options);
222 
223 		if (!node.nodeType) {
224 			if (typeof node == 'string') {
225 				node = {
226 					template: node
227 				};
228 			}
229 			var data = {};
230 			Object.keys(self._defaultOptions).forEach(function(key) {
231 				if (options[key] === undefined) data[key] = self.get(key);
232 			});
233 			extend(data, options);
234 
235 			var tdata;
236 			if (node.section) {
237 				tdata = {};
238 				tdata[node.section] = data;
239 			} else {
240 				tdata = data;
241 			}
242 			var str = string.substitute(node.template, tdata);
243 			node = dom.Element.fromString(str);
244 		}
245 
246 		self._node = dom.wrap(node);
247 
248 		self.__initEvents();
249 		self.__initOptions(options);
250 		self.__initSubs();
251 	};
252 
253 	/**
254 	 * 加入 _eventType 方法定义的事件
255 	 */
256 	this.__initEvents = function(self) {
257 		self._events.forEach(function(eventType) {
258 			self.addEvent(eventType, function() {
259 				self['_' + eventType].apply(self, arguments);
260 			});
261 		});
262 	};
263 
264 	this.__initOptions = function(self, options) {
265 		Object.keys(self._defaultOptions).forEach(function(name) {
266 			// 从dom获取配置
267 			var data = self._node.getData(name.toLowerCase()),
268 			defaultValue = self._defaultOptions[name],
269 			value;
270 
271 			if (data) {
272 				value = getConstructor(typeof defaultValue)(data);
273 				self.__setOption(name, value);
274 			// 从options参数获取配置
275 			} else if (options[name]) {
276 				self.__setOption(name, options[name]);
277 			// 默认配置
278 			} else {
279 				self.__setOption(name, defaultValue);
280 			}
281 		});
282 	};
283 
284 	this.__initSubs = function(self) {
285 		Object.keys(self._subs).forEach(function(name) {
286 			var sub = self._subs[name];
287 			var node;
288 
289 			if (sub.single) {
290 				node = self._node.getElement(sub.selector);
291 			} else {
292 				node = self._node.getElements(sub.selector);
293 			}
294 
295 			self.__initSub(name, node);
296 		});
297 	};
298 
299 	/**
300 	 * 根据sub的定义获取component的引用
301 	 */
302 	this.__initSub = function(self, name, nodes) {
303 		if (!self._node) return null;
304 
305 		var sub = self._subs[name];
306 		var comps;
307 
308 		if (sub.single) {
309 			if (nodes) {
310 				comps = new sub.type(nodes, self._subOptions[name]);
311 				self.__fillSub(name, comps);
312 			}
313 		} else {
314 			if (nodes) {
315 				comps = new exports.Components(nodes, sub.type, self._subOptions[name]);
316 				comps.forEach(function(comp) {
317 					self.__fillSub(name, comp);
318 				});
319 			} else {
320 				// 没有的也留下一个空的Components
321 				comps = new exports.Components([], sub.type);
322 			}
323 		}
324 
325 		self['_' + name] = nodes;
326 		self._set(name, comps);
327 
328 		return comps;
329 	};
330 
331 	/**
332 	 * 将一个comp的信息注册到_subs上
333 	 */
334 	this.__fillSub = function(self, name, comp) {
335 		var sub = self._subs[name];
336 		var node = comp._node;
337 		sub.nodeMap[String(node.uid)] = comp;
338 		self.__addEventTo(name, node);
339 	};
340 
341 	this.getOption = function(self, name) {
342 		var pname = '_' + name;
343 		if (self[pname] === undefined) {
344 			self[pname] = self._defaultOptions[name];
345 		}
346 		return self[pname];
347 	};
348 
349 	/**
350 	 */
351 	this.setOption = options.overloadsetter(function(self, name, value) {
352 		var parts = Array.isArray(name)? name : name.split('.');
353 		if (parts.length > 1) {
354 			self.__setSubOption(parts, value);
355 			if (self[parts[0]]) {
356 				self[parts[0]].setOption(parts.slice(1), value);
357 			}
358 		} else {
359 			var pname = '_' + name;
360 			var methodName = name + 'Change';
361 			self.__setOption(name, value);
362 			if (self[methodName]) self[methodName](value);
363 		}
364 	});
365 
366 	this.__setOption = function(self, name, value) {
367 		var pname = '_' + name;
368 		self[pname] = value;
369 		self._set(name, value);
370 	};
371 
372 	this.__addEventTo = function(self, name, node) {
373 		var events = self._subEvents[name];
374 		if (!events) return;
375 
376 		events.forEach(function(eventType) {
377 			node.addEvent(eventType, function() {
378 				var args = Array.prototype.slice.call(arguments, 0);
379 				var comp = self._subs[name].nodeMap[String(node.uid)];
380 				args.push(comp); // 最后一个参数为component
381 				self[name + '_' + eventType].apply(self, args);
382 			});
383 		});
384 	};
385 
386 	/**
387 	 * 向_subOptions生成一个option
388 	 * {'a.b.c': 1, b: 2} ==> {a: {b: {c:1}}, b: 2}
389 	 */
390 	this.__setSubOption = function(self, name, value) {
391 		var current = self._subOptions;
392 		var parts = Array.isArray(name)? name : name.split('.');
393 		// 生成前缀对象
394 		for (var i = 0, part; i < parts.length - 1; i++) {
395 			part = parts[i];
396 			if (current[part] === undefined) {
397 				current[part] = {PARSED: true};
398 			}
399 			current = current[part];
400 		}
401 		current[parts[parts.length - 1]] = value;
402 	};
403 
404 	/**
405 	 * 初始化 subOptions
406 	 */
407 	this.__initSubOptions = function(self, options) {
408 		if (!options.PARSED) {
409 			self._subOptions = {PARSED: true};
410 			Object.keys(options).forEach(function(name) {
411 				self.__setSubOption(name, options[name]);
412 			});
413 		} else {
414 			self._subOptions = options;
415 		}
416 	};
417 
418 	/**
419 	 * 渲染一组subcomponent
420 	 * @param name subcomponent名字
421 	 * @param data 模板数据/初始化参数
422 	 */
423 	this.render = function(self, name, data) {
424 
425 		var sub = self._subs[name];
426 		var methodName = 'render' + string.capitalize(name);
427 		var method2Name = name + 'Render';
428 		var nodes;
429 
430 		// 如果已经存在结构了，则不用再render了
431 		// 没有render方法，则返回
432 		if (!!(sub.single? self[name] : self[name].length) || (!self[methodName] && !self[method2Name])) {
433 			return;
434 		}
435 
436 		if (self[method2Name]) {
437 			nodes = self[method2Name](function() {
438 				return self.make(name, data);
439 			});
440 		} else {
441 			nodes = self[methodName](data);
442 		}
443 
444 		// 如果有返回结果，说明没有使用self.make，而是自己生成了需要的普通node元素，则对返回结果进行一次包装
445 		if (nodes) {
446 			if (sub.single) {
447 				if (Array.isArray(nodes)) throw '这是一个唯一引用元素，请不要返回一个数组';
448 				sub.rendered.push(nodes);
449 			} else {
450 				if (!Array.isArray(nodes)) throw '这是一个多引用元素，请返回一个数组';
451 				nodes = new dom.Elements(nodes);
452 				sub.rendered = sub.rendered.concat(nodes);
453 			}
454 
455 			self.__initSub(name, nodes);
456 		}
457 	};
458 
459 	/**
460 	* 根据subs的type创建一个component，并加入到引用中，这一般是在renderXXX方法中进行调用
461 	* @param name
462 	* @param data 模板数据
463 	*/
464 	this.make = function(self, name, data) {
465 		var sub = self._subs[name];
466 		var pname = '_' + name;
467 		var options = {};
468 		var extendOptions = self._subOptions[name];
469 		extend(options, extendOptions);
470 
471 		if (data) {
472 			Object.keys(data).forEach(function(key) {
473 				options[key] = data[key];
474 			});
475 		}
476 
477 		var comp = new sub.type({
478 			template: sub.template,
479 			section: sub.section
480 		}, options);
481 		var node = comp._node;
482 
483 		if (sub.single) {
484 			self[name] = comp;
485 			self[pname] = node;
486 		} else {
487 			self[name].push(comp);
488 			self[pname].push(node);
489 		}
490 		self.__fillSub(name, comp);
491 		sub.rendered.push(node);
492 
493 		return comp;
494 	};
495 
496 	/**
497 	 * 设置subcomponent的template
498 	 */
499 	this.setTemplate = function(self, name, template, section) {
500 		self._subs[name].template = template;
501 		self._subs[name].section = section;
502 	};
503 
504 	/**
505 	 * 弹出验证错误信息
506 	 */
507 	this.invalid = fireevent(function(self, msg) {
508 		if (!msg) msg = '输入错误';
509 		alert(msg);
510 	});
511 
512 	/**
513 	 * 弹出出错信息
514 	 */
515 	this.error = fireevent(function(self, msg) {
516 		if (!msg) msg = '出错啦！';
517 		alert(msg);
518 	});
519 
520 	/**
521 	 * 重置一个component，回到初始状态，删除所有render的元素。
522 	 */
523 	this.reset = fireevent(function(self) {
524 		// 清空所有render进来的新元素
525 		Object.keys(self._subs).forEach(function(name) {
526 			var sub = self._subs[name];
527 			var pname = '_' + name;
528 			sub.rendered.forEach(function(node) {
529 				var comp = sub.nodeMap[node.uid];
530 				delete sub.nodeMap[node.uid];
531 				node.dispose();
532 				if (sub.single) {
533 					self[name] = self[pname] = null;
534 				} else {
535 					self[name].splice(self[name].indexOf(comp), 1); // 去掉
536 					self[pname].splice(self[pname].indexOf(node), 1); // 去掉
537 				}
538 			});
539 			if (!sub.single) {
540 				self[name].forEach(function(comp) {
541 					comp.reset();
542 				});
543 			} else if (self[name]) {
544 				self[name].reset();
545 			}
546 		});
547 	});
548 
549 	/**
550 	 * 获取包装的节点
551 	 */
552 	this.getNode = function(self) {
553 		return self._node;
554 	};
555 
556 	this.define = staticmethod(exports.define);
557 	this.define1 = staticmethod(exports.define1);
