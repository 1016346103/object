1 /**
  2  * @namespace
  3  * @name ui.nav
  4  */
  5 object.add('ui.nav', 'ui, ui.decorators', /**@lends ui.nav*/ function(exports, ui) {
  6 
  7 var fireevent = ui.decorators.fireevent;
  8 
  9 /**
 10  * @class
 11  * @name ui.nav.ForeNextControl
 12  */
 13 this.ForeNextControl = new Class(ui.Component, /**@lends ui.nav.ForeNextControl*/ function() {
 14 
 15 	this.nextButton = ui.define('.nextbutton');
 16 	this.foreButton = ui.define('.forebutton');
 17 
 18 	this.initialize = function(self, node) {
 19 		ui.Component.initialize(self, node);
 20 
 21 		self.loop = false; // 是否循环
 22 		self.total = parseInt(self._node.getData('total'));
 23 		self.start = parseInt(self._node.getData('start')) || 0;
 24 		self.position = self.start;
 25 	};
 26 
 27 	this.nextButton_click = function(self, event) {
 28 		if (self.position >= self.total - 1) {
 29 			if (self.loop) self.position = -1;
 30 			else return;
 31 		}
 32 		self.next();
 33 	};
 34 
 35 	this.foreButton_click = function(self, event) {
 36 		if (self.position <= 0) {
 37 			if (self.loop) self.position = self.total;
 38 			else return;
 39 		}
 40 		self.fore();
 41 	};
 42 
 43 	this.next = fireevent(function(self) {
 44 		self.position++;
 45 		self.change();
 46 	});
 47 
 48 	this.fore = fireevent(function(self) {
 49 		self.position--;
 50 		self.change();
 51 	});
 52 
 53 	this.change = function(self) {
 54 		// TODO
 55 		self.fireEvent('change', {forenext: self});
 56 		self.updateTotal();
 57 		self.updatePosition();
 58 	};
 59 
 60 	this.updatePosition = fireevent(function(self) {
 61 		self._node.getElements('.current').set('innerHTML', self.position + 1); // position是从0开始滴～展示的时候+1
 62 	});
 63 
 64 	this.updateTotal = fireevent(function(self) {
 65 		self._node.getElements('.total').set('innerHTML', self.total);
 66 	});
 67 
 68 });
 69 
 70 });
