  3  * @name ua.os

  5 object.add('ua.os', 'sys', function(exports, sys) {
  6 
  7 var uamodule = sys.modules['ua'];
  8 
  9 //由于需要先替换下划线，与ua模块中的numberify不同，因此这里再定义此方法
 10 var numberify = function(s) {
 11 	var c = 0;
 12 	// convert '1.2.3.4' to 1.234
 13 	return parseFloat(s.replace(/_/g, '.').replace(/\./g, function() {
 14 		return (c++ === 0) ? '.' : '';
 15 	}));
 16 };
 17 
 18 if (uamodule) {
 19 	//将detectOS方法导出，便于单元测试
 20 	this._detectOS = detectOS;
 21 	var o = detectOS(navigator.userAgent.toLowerCase());
 22 	object.extend(uamodule.ua, o);
 23 }

 25 //判断对象obj是否是type类型
 26 function is(obj, type) {
 27 	type = type.replace(/\b[a-z]/g, function(match){
 28 		return match.toUpperCase();
 29 	});
 30 	return Object.prototype.toString.call(obj) == '[object ' + type + ']';
 31 }
 32 
 33 //断言，如果bool不是true，则抛出异常消息msg
 34 function assertTrue(bool, msg) {
 35 	if(!bool) {
 36 		throw new Error(msg);
 37 	}
 38 }
 39 
 40 //断言，确保传入的obj不是空，如果为空，则抛出异常消息msg
 41 function assertNotNull(obj, msg) {
 42 	if(obj == null) {
 43 		throw new Error(msg);
 44 	}
 45 }

 47 //传入ua，便于模拟ua字符串进行单元测试
 48 //
 49 //http://forums.precentral.net/palm-pre-pre-plus/277613-webos-2-1-user-agent.html
 50 //what is the relationship between webos and palmos????
 51 //http://www.developer.nokia.com/Community/Wiki/User-Agent_headers_for_Nokia_devices
 52 //how to handle the NokiaXXXX?
 53 function detectOS(ua) {
 54 	ua = ua || navigator.userAgent;
 55 	ua = ua.toLowerCase();
 56 	
 57 	/**
 58 	 * 所有的操作系统检测的配置项
 59 	 *	{
 60 	 *		core: 操作系统内核
 61 	 *		match: 操作系统内核匹配，可以是正则表达式，也可以是function，function参数是userAgent字符串，返回值是true/false
 62 	 *		versionRule：获取操作系统版本的正则表达式
 63 	 *		version: 指定的操作系统版本值
 64 	 *	} 
 65 	 */
 66 	var osDetecters = [
 67 	{core: 'windowsnt',		match: function(ua) {
 68 								return /windows\snt/.test(ua) && !/xblwp7/.test(ua);
 69 							},						versionRule: /windows nt\s([\.\d]*)/},
 70 	{core: 'windowsnt',		match: /windows\sxp/,	version: 5.1},
 71 	{core: 'windowsnt', 	match: /windows\s2000/, version: 5.0},
 72 	{core: 'windowsnt', 	match: /winnt/,			version: 4.0},
 73 	{core: 'windows',		match: /windows me/,	version: 'me'},
 74 	{core: 'windows',		match: /windows 98|win98/,version: '98'},
 75 	{core: 'windows',		match: /windows 95|win95/,version: '95'},
 76 	{core: 'windows',		match: /win16/,			version: '3.1'},
 77 	{core: 'windows/phone',	match: /windows\sphone/,versionRule: /windows phone os ([\d\.]*)/},
 78 	{core: 'windows/phone',	match: /xblwp7/,		version: 7.0},
 79 	{core: 'windows/mobile',match: /windows mobile|wce|windows ce|pocket pc|wince/,	
 80 													versionRule: /iemobile ([\.\d]*)/},
 81 	{core: 'windows',		match: /win/,			version: 'unknown'},
 82 	
 83 	{core: 'android', 		match: /\sandroid/,		versionRule:/android ([^\s]*);/},
 84 
 85 	{core: 'linux/debian',	match: /debian/, 		versionRule: /debian[\s\/-]([\.\d]*)/},
 86 	{core: 'linux/redhat',	match: /red\shat/, 		versionRule: /red hat[\s\/-]([\.\d]*)/},
 87 	{core: 'linux/fedora',	match: /fedora/, 		versionRule: /fedora[\s\/-]([\.\d]*)/},
 88 	{core: 'linux/ubuntu',	match: /ubuntu/, 		versionRule: /ubuntu[\s\/-]([\.\d]*)/},
 89 	{core: 'linux/suse',	match: /suse/, 			versionRule: /suse[\s\/-]([\.\d]*)/},
 90 	{core: 'linux/mint',	match: /mint/, 			versionRule: /mint[\s\/-]([\.\d]*)/},
 91 	{core: 'linux/centos',	match: /centos/, 		versionRule: /centos[\s\/-]([\.\d]*)/},
 92 	{core: 'linux/gentoo',	match: /gentoo/, 		version: 'unknown'},
 93 	{core: 'linux',			match: /linux/,			version: 'unknown'},
 94 
 95 	{core: 'chromeos' ,		match: /cros/,  		version: 'unknown'},
 96 
 97 	{core: 'unix/sunos' ,	match: /sunos/,  		version: 'unknown'},
 98 	{core: 'unix/freebsd',	match: /freebsd/,  		version: 'unknown'},
 99 	{core: 'unix/openbsd',	match: /openbsd/,  		version: 'unknown'},
100 	{core: 'unix/aix' ,		match: /aix/,  			version: 'unknown'},
101 	{core: 'unix/hp_ux' ,	match: /hp-ux/,  		version: 'unknown'},
102 	{core: 'unix',			match: /x11/,			version: 'unknown'},
103 	
104 	{core: 'macos' ,		match:/mac_powerpc|ppc/,version: 'ppc'},
105 	{core: 'macos' ,		match: /intel/,  		version: 'intel'},
106 	{core: 'macos' ,		match: /mac_68000|68k/, version: '68k'},
107 	{core: 'ios',			match: function(ua) {
108 		   						return /applewebkit/.test(ua) && / mobile\//.test(ua) && /like/.test(ua);
109 	   						},						versionRule: /os ([\_\.\d]*)/},
110 	{core: 'macos' ,		match: /mac/,  			version: 'unknown'},
111 	
112 	{core: 'os2' ,			match: function(ua) {
113 								return /os\/2|ibm-webexplorer/.test(ua) || navigator.appVersion.indexOf("os/2") != -1;
114 							},						version: 'unknown'},
115 	{core: 'symbian',		match: /symbian|s60|symbos|symbianos|series40|series60|nokian/,
116 													versionRule: /symbian(?:os)?\/([\d\.]*);/},
117 	{core: 'blackberry',	match: /blackberry|rim\stablet\sos/, 					
118 													versionRule: /(?:version\/|blackberry[\d]{4}\/)([\d\.]*)/},
119 	{core: 'webos', 		match: /webos/,			versionRule:/webos\/([^\s]*);/},
120 	{core: 'palmos',		match: /palmos/,		version: 'unknown'}
121 	];
122 
123 	var o = {};
124 
125 	//操作系统检测主逻辑
126 	for(var i=0, l=osDetecters.length, current, matchFlag = false; i<l; i++) {
127 		current = osDetecters[i];
128 		var match = current.match;
129 		//确保match是正则表达式或者是function
130 		assertTrue(is(match, 'RegExp') || is(match, 'Function'), 'match rule should be regexp or function');
131 		if(is(match, 'RegExp')) {
132 			//如果是正则表达式，则查看是否匹配
133 			matchFlag = match.test(ua);
134 		}else if(is(match, 'Function')) {
135 			//如果是方法，则执行，并传入ua作为参数
136 			matchFlag = match(ua);
137 			assertNotNull(matchFlag, 'match function must return true/false');
138 		} 
139 		//如果不匹配，则继续循环
140 		if(!matchFlag) {
141 			continue;
142 		}
143 		//执行到这里，说明已经匹配了
144 		var parent=null, packages=current.core.split('\/'), pLength=packages.length;
145 		if(pLength > 1) {
146 			//说明有子类型，比如windows/phone
147 			o.oscore = packages[0];
148 			parent = o;
149 			//构造子类型对象链
150 			for(var m=0; m<pLength - 1; m++) {				
151 				parent = parent[packages[m]] = {};
152 			}
153 		} else {
154 			o.oscore = current.core;
155 		}
156 		//获取版本信息
157 		var version = current.version || 'unknown';
158 		//如果有版本获取规则，则执行此规则，规则中必须取出版本号
159 		if(current.versionRule) {
160 			assertTrue(is(current.versionRule, 'RegExp'), 'version rule should be regexp');
161 			m = ua.match(current.versionRule);
162 			if(m && m[1]) version = numberify(m[1]);
163 		}
164 		//将版本信息放入返回的对象中
165 		if(parent) {
166 			parent[packages[pLength - 1]] = version;
167 		} else {
168 			o[o.oscore] = version;
169 		}
170 		break;
171 	}
172 	
173 	//如果是ios，继续判断移动设备
174 	if(o.ios) {
175 		m = ua.match(/ipad|ipod|iphone/);
176 		if (m && m[0]) {
177 			o[m[0]] = o.ios;
178 		}
179 	}
180 	//判断 Google Caja, from YUI-client
181 	if(navigator && navigator.cajaVersion) {
182 		o.caja = navigator.cajaVersion;
183 	}
184 
185 	if(!matchFlag) {
186 		o.oscore = 'unknown';
187 	}
188 
189 	//wow64  : Windows-On-Windows 64-bit
190 	//x64    : 64-bit windows version
191 	//win64  : Win32 for 64-Bit-Windows
192 	//ia64   : I-tanium 64-bit processor from Intel
193 	//sparc64: 64-bit Sun UltraSPARC processor
194 	//ppc64  : 64-bit PowerPC microprocessor
195 	//x86_64 : 64-bit Intel processor
196 	if (/wow64|x64|win64|ia64|x86_64|amd64|sparc64|ppc64/.test(ua)){
197 		o.processor = 64;
198 	} else {
199 		o.processor = 32;
200 	}
201 	
202 	//检测分辨率（devicePixelRatio说明是高密度的显示屏，如iphone）
203 	//http://developer.android.com/guide/webapps/targeting.html
204 	if(window.devicePixelRatio >= 2) {
205 		o.resolution = {
206 			width : screen.width  * window.devicePixelRatio,
207 			height: screen.height * window.devicePixelRatio
208 		};
209 	} else {
210 		o.resolution = {
211 			width: screen.width,
212 			height: screen.height
213 		}
214 	}
215 
216 	//检测屏幕方向，首先确保支持屏幕方向
217 	var isMobile = typeof window.orientation != 'undefined' ? true : false;
218 	if(isMobile) {
219 		//通过屏幕的高度和宽度的值大小，来判断是横向还是纵向
220 		o.orientation = window.innerWidth > window.innerHeight ? 'profile' : 'landscape';
221 	} else {
222 		o.orientation = 'unknown';
223 	}
224 
225 	return o;
226 }
227 });
