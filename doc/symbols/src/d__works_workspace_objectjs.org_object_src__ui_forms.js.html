1 object.add('ui.forms', 'dom, ui, ui.decorators, ua', function(exports, dom, ui, ua) {
  2 
  3 var fireevent = ui.decorators.fireevent;
  4 
  5 this.SelectionInputValues = new Class(Array, function() {
  6 
  7 	this.initialize = function(self, input) {
  8 		self._input = input;
  9 	};
 10 
 11 	this.push = function(self, value) {
 12 		this.parent(self, value);
 13 
 14 		self._input.setValues(self);
 15 	};
 16 
 17 	this.pop = function(self) {
 18 		var value = this.parent(self);
 19 
 20 		self._input.setValues(self);
 21 		return value;
 22 	};
 23 
 24 	this.splice = function(self, index, length) {
 25 		var value = this.parent(self, index, length);
 26 
 27 		self._input.setValues(self);
 28 		return value;
 29 	};
 30 
 31 });
 32 
 33 this.SelectionInput = new Class(ui.Component, function() {
 34 
 35 	this.maxCount = ui.option(-1);
 36 
 37 	this.noRepeat = ui.option(false);
 38 
 39 	this.initialize = function(self, node, options) {
 40 		this.parent(self, node, options);
 41 
 42 		self.name = self._node.name;
 43 		self.values = new exports.SelectionInputValues(self);
 44 
 45 		// Webkit 不能通过设置input的padding影响其光标位置（永远在中间），因此将self._node替换成一个textarea
 46 		// @see: https://bugs.webkit.org/show_bug.cgi?id=32981
 47 		if (ua.ua.webkit) {
 48 			var textarea = dom.wrap(document.createElement('textarea'));
 49 			textarea.style.resize = 'none';
 50 			textarea.style.overflow = 'hidden';
 51 			textarea.rows = 1;
 52 			for (var i = 0; i < self._node.attributes.length; i++) {
 53 				var attribute = self._node.attributes[i];
 54 				textarea.setAttribute(attribute.name, attribute.value);
 55 			}
 56 			self._node.parentNode.replaceChild(textarea, self._node);
 57 			self._node = textarea;
 58 		}
 59 
 60 		// 去掉输入框的value避免提交空白value上去
 61 		self._node.removeAttribute('name');
 62 		self._node.addEvent('keydown', function(event) {
 63 			if (event.keyCode === 8) {
 64 				if (!self._node.get('value').trim()) {
 65 					event.preventDefault();
 66 					self.values.pop();
 67 				}
 68 			} else if (event.keyCode === 188/*,*/ || event.keyCode === 59 /*;*/ || event.keyCode === 13 /*return*/ || event.keyCode === 32/*space*/) {
 69 				event.preventDefault();
 70 				self.__inputValue();
 71 			}
 72 		});
 73 
 74 		self._node.addEvent('blur', function(event) {
 75 			self.__inputValue();
 76 		});
 77 	};
 78 
 79 	this.__inputValue = function(self, value) {
 80 		var value = self._node.get('value').trim();
 81 		if (value && self._node.checkValidity()) {
 82 			self.addValue(value);
 83 		} else {
 84 			self._node.set('value', '');
 85 		}
 86 	};
 87 
 88 	this.addValue = fireevent(function(self, value) {
 89 		if ((self.maxCount !== -1 && self.values.length >= self.maxCount)
 90 			|| (self.noRepeat && self.values.indexOf(value) !== -1)) {
 91 			self._node.set('value', '');
 92 		} else {
 93 			self.values.push(value);
 94 		}
 95 	});
 96 
 97 	//this.addValue_dataPasser = function(self, value) {
 98 		//return {value: value};
 99 	//};
100 
101 	this.makeWrapper = function(self) {
102 		if (self.selectionWrapper) return;
103 
104 		var style = window.getComputedStyle? getComputedStyle(self._node, null) : self._node.currentStyle;
105 
106 		self.paddingTop = parseInt(style.paddingTop);
107 		self.paddingLeft = parseInt(style.paddingLeft);
108 		self.paddingBottom = parseInt(style.paddingBottom);
109 		self.paddingRight = parseInt(style.paddingRight);
110 		self.top = parseInt(style.borderTopWidth) + self.paddingTop;
111 		self.left = parseInt(style.borderLeftWidth) + self.paddingLeft;
112 		self.bottom = parseInt(style.borderBottomWidth) + self.paddingBottom;
113 		self.right = parseInt(style.borderRightWidth) + self.paddingRight;
114 
115 		// css 必须设置line-height，否则在webkit中取到的是 normal
116 		self.oneLineHeight = parseInt(style.lineHeight);
117 		//console.log('line height', self.oneLineHeight);
118 
119 		// 包装
120 		var selectionWrapper = document.createElement('div');
121 		selectionWrapper.className = 'selectioninputwrapper';
122 		selectionWrapper.style.width = (self._node.offsetWidth - self.left - self.right) + 'px';
123 		selectionWrapper.style.marginTop = self.top + 'px';
124 		selectionWrapper.style.marginLeft = self.left + 'px';
125 		var inner = document.createElement('span');
126 		selectionWrapper.appendChild(inner);
127 		self._node.parentNode.insertBefore(selectionWrapper, self._node);
128 		self.selectionWrapper = inner;
129 	};
130 
131 	/**
132 	 * 生成一个
133 	 */
134 	this.makeSelection = function(self, value) {
135 		var input = document.createElement('input');
136 		input.name = self.name;
137 		input.type = 'hidden';
138 		input.value = value;
139 		var selection = document.createElement('span');
140 		selection.className = 'selectioninputselection';
141 		var deleteButton = document.createElement('span');
142 		deleteButton.innerHTML = '×';
143 		deleteButton.className = 'selectioninputdelete';
144 		deleteButton.onclick = function() {
145 			self.values.splice(parseInt(this.parentNode.getAttribute('data-index')), 1);
146 		};
147 		selection.innerHTML = value;
148 		selection.appendChild(deleteButton);
149 		selection.appendChild(input);
150 		return selection;
151 	};
152 
153 	this.setValues = function(self, values) {
154 
155 		self.makeWrapper();
156 		while (self.selectionWrapper.firstChild) self.selectionWrapper.removeChild(self.selectionWrapper.firstChild);
157 
158 		values.forEach(function(value, i) {
159 			var selection = self.makeSelection(value);
160 			selection.setAttribute('data-index', i);
161 			self.selectionWrapper.appendChild(selection);
162 			var style = window.getComputedStyle? getComputedStyle(selection, null) : selection.currentStyle;
163 			var topWidth = parseInt(style.marginTop) + parseInt(style.borderTopWidth) + parseInt(style.paddingTop);
164 			var bottomWidth = parseInt(style.marginBottom) + parseInt(style.borderBottomWidth) + parseInt(style.paddingBottom);
165 			var leftWidth = parseInt(style.marginLeft) + parseInt(style.borderLeftWidth) + parseInt(style.paddingLeft);
166 			var rightWidth = parseInt(style.marginRight) + parseInt(style.borderRightWidth) + parseInt(style.paddingRight);
167 			var height = (self.oneLineHeight - topWidth - bottomWidth);
168 			selection.style.height = height + 'px';
169 		});
170 
171 		var rects = self.selectionWrapper.getClientRects();
172 		var lineCount, lastLine, lastLineWidth;
173 		// IE下不能通过getClientRects获取行数，只能逐个便利，通过top坐标的变化判断行数
174 		if (ua.ua.ie) {
175 			lineCount = 0;
176 			var last = 0, lastLineFirstRect;
177 			for (var i = 0; i < rects.length; i++) {
178 				var rect = rects[i];
179 				if (rect.top != last) { // 新行的第一个元素
180 					lastLineFirstRect = rect;
181 					lineCount++;
182 				}
183 				last = rect.top;
184 			}
185 			lastLineWidth = rect.right - lastLineFirstRect.left;
186 		} else {
187 			lineCount = rects.length;
188 			// 最后一行的宽度
189 			lastLine = rects[rects.length - 1];
190 			lastLineWidth = parseInt(lastLine.right - lastLine.left);
191 		}
192 
193 		var wrapperWidth = parseInt(self.selectionWrapper.parentNode.style.width);
194 
195 		if (lastLineWidth >= wrapperWidth/* * 0.8*/) { // 本行溢出了
196 			self.width = (self._node.offsetWidth - self.left - self.right);
197 			self._node.style.width = self.width + 'px';
198 			self._node.style.paddingLeft = self.paddingLeft + 'px';
199 			self._node.style.paddingTop = (self.oneLineHeight * lineCount + self.paddingTop) + 'px';
200 		} else {
201 			self.width = (self._node.offsetWidth - lastLineWidth - self.left - self.right);
202 			self._node.style.width = self.width + 'px';
203 			self._node.style.paddingLeft = lastLineWidth + self.paddingLeft + 'px';
204 			self._node.style.paddingTop = (self.oneLineHeight * (lineCount - 1) + self.paddingTop) + 'px';
205 		}
206 
207 		if (values.length) {
208 			self.placeholder = self._node.getAttribute('placeholder');
209 			self._node.removeAttribute('placeholder');
210 			self._node.set('value', '');
211 		}
212 	};
213 
214 });
215
