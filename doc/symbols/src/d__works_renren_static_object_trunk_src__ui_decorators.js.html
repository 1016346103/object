  5 object.add('ui.decorators', /**@lends ui.decorators*/ function(exports, string, dom) {
  6 
  7 	/**
  8 	 * fireevent 或 fireevent(eventName)
  9 	 * fireevent 默认eventName通过__name__获得
 10 	 */
 11 	this.fireevent = function(func) {
 12 		// 自定义了事件名称，返回一个decorator
 13 		if (typeof func == 'string') {
 14 			var name = func;
 15 			return function(func) {
 16 				var result = function(self) {
 17 					if (self.nofireevent !== true) self._node.fireEvent(name, null, self);
 18 					func.apply(this, arguments);
 19 				};
 20 				return result;
 21 			};
 22 		} else {
 23 			return function(self) {
 24 				if (self.nofireevent !== true) self._node.fireEvent(arguments.callee.__name__, null, self);
 25 				func.apply(this, arguments);
 26 			};
 27 		}
 28 	};
 29 
 30 });
