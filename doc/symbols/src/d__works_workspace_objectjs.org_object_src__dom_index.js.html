114 		return new Elements(node);

116 		// 已经wrap过了
117 		if (node._wrapped) return node;

120 		if (node === window) {

141 /**
142  * 通过selector获取context作用域下的节点集合
143  * dom.Elements包装后的节点数组拥有相应最小Element的统一调用方法
144  * 比如 forms = dom.getElements('form'); 'send' in forms // true
145  * @function
146  * @name dom.getElements
147  * @param selector 一个css selector
148  * @param context 一个节点
149  * @returns {dom.Elements}
150  */
151 var getElements = this.getElements = function(selector, context) {
152 	if (!context) context = document;
153 
154 	// 解析成Slick Selector对象
155 	var parsed = Slick.parse(selector);
156 
157 	// Slick在面对自定义标签时各种不靠谱，换用sizzle
158 	var eles = Sizzle(selector, context);
159 
160 	// 这里通过分析selector的最后一个部分的tagName，来确定这批eles的wrapper
161 	// 例如selector是 div form.xxx 则wrapper是 FormElement
162 	// 例如selector是 div .xxx 则wrapper是 Element
163 	// 例如selector是 div select.xxx, div.input.xxx 则wrapper是 FormItemElement
164 
165 	var wrapper, part;
166 	// 绝大部分情况都是length=0，只有1个selector，保证其性能
167 	if (parsed.expressions.length == 1) {
168 		part = parsed.expressions[0];
169 		wrapper = getWrapper(part[part.length - 1].tag);
170 
171 	// 由多个selector组成，比如 div select.xxx, div.input.xxx，要保证这种能取到 FormItemElement
172 	} else {
173 		// 通过生成每个selector wrapper的继承链，不断的生成当前selector和上一个selector的继承链的相同部分
174 		// 最后的chain的最后一个元素，既是公用wrapper
175 		for (var i = 0, chain, previousChain; i < parsed.expressions.length; i++) {
176 			part = parsed.expressions[i];
177 			wrapper = getWrapper(part[part.length - 1].tag);
178 
179 			// 当前selector最后元素的wrapper chain
180 			// slice(0, -1) 过滤掉Element继承的 Attribute 类
181 			chain = Class.getChain(wrapper).slice(0, -1).reverse();
182 			if (previousChain) {
183 				chain = getCommon(chain, previousChain);
184 			}
185 			// 如果相同部分length=1，则代表找到Element类了，可以停止继续搜索
186 			if (chain.length == 1) break;
187 			previousChain = chain;
188 		}
189 		wrapper = chain[chain.length - 1];
190 	}
191 
192 	return new Elements(eles, wrapper);
193 };

306 function isEventSupported(eventName, element) {
307 	var TAGNAMES = {
308 		'select': 'input', 'change': 'input',
309 		'submit': 'form', 'reset': 'form',
310 		'error': 'img', 'load': 'img', 'abort': 'img'
311 	};
312 	element = element || document.createElement(TAGNAMES[eventName] || 'div');
313 	eventName = 'on' + eventName;
314 	
315 	var isSupported = (eventName in element);
316 	
317 	if (!isSupported) {
318 		// if it has no `setAttribute` (i.e. doesn't implement Node interface), try generic element
319 		if (!element.setAttribute) {
320 			element = document.createElement('div');
321 		}
322 		if (element.setAttribute && element.removeAttribute) {
323 			element.setAttribute(eventName, '');
324 			isSupported = typeof element[eventName] == 'function';
325 
326 			// if property was created, "remove it" (by setting value to `undefined`)
327 			if (typeof element[eventName] != 'undefined') {
328 				element[eventName] = undefined;
329 			}
330 			element.removeAttribute(eventName);
331 		}
332 	}
333 	
334 	element = null;
335 	return isSupported;
336 }

680 	/**
681 	 * 根据鼠标的移动距离，判断是否已经开始拖拽
682 	 *
683 	 * 初始情况下为document的mousemove方法添加的是checkDragging，判断是否是拖拽操作
684 	 * 如果开始拖拽，再将checkDragging改为dragging，正式执行拖拽的功能

686 	 * @param e 事件对象

688 	this._checkDragging = function(self, e) {
689 		//在IE下，如果拖动非常迅速时，鼠标变成禁止符号，这里需要禁止默认事件的发生
690 		if(e.preventDefault) e.preventDefault();
691 		
692 		//计算鼠标移动的距离，如果大于某一个阈值，则认为开始拖动
693 		//这是Mootools的方案，Kissy还提供了一种鼠标点击持续事件的判断，如果大于200ms，说明是拖拽
694 		var mousePos = getMousePos(e);
695 		var distance = Math.round(Math.sqrt(Math.pow(mousePos.x - self.__originMouseX, 2) + 
696 				Math.pow(mousePos.y - self.__originMouseY, 2)));
697 		//说明开始拖拽了
698 		if(distance > 3) {
699 			//把mousemove由检查拖拽改为执行拖拽，把mouseup由取消改为完成
700 			self._removeEventFromDoc('mousemove', self.__binderForDD.checkDragging, false);
701 			self._removeEventFromDoc('mouseup', self.__binderForDD.cancel, false);
702 			self._addEventToDoc('mousemove', self.__binderForDD.dragging, false);
703 			self._addEventToDoc('mouseup', self.__binderForDD.finish, false);
704 		
705 			//给元素添加拖拽时候的基本样式
706 			addDraggingStyle(self);

708 			//触发dragstart事件，参考HTML5规范
709 			self.fireEvent('dragstart', {dragging:self, event:e});

711 			//这里也触发所属元素的dropinit事件
712 			//dropinit不是HTML5规范规定的，但是也是有必要的
713 			//dragstart, drag, dragend是draggable元素的完整生命周期，
714 			//但是如果没有dropinit，droppable元素只有dropenter, dropover, dropleave, drop，没有初始状态，不完整
715 			//具体示例：如果在拖拽初始时需要创建占位元素，如果没有dropinit，就只能针对每一个元素的dragstart编写代码了
716 			if(self.__belongToDroppable) {
717 				self.__belongToDroppable.fireEvent('dropinit', {dragging:self, event:e});
718 			}
719 		}

722 	/**
723 	 * 拖拽时的事件处理方法
724 	 *
725 	 * @param e 事件对象
726 	 */
727 	this._dragging = function(self, e) {
728 		//阻止默认事件
729 		if(e.preventDefault) e.preventDefault();
730 
731 		//利用鼠标位置，修改拖拽元素的位置
732 		var mousePos = getMousePos(e);
733 		self.style.left = (mousePos.x - self.__deltaX) + 'px';
734 		self.style.top  = (mousePos.y - self.__deltaY) + 'px';
735 		//触发drag事件，遵循HTML5规范
736 		self.fireEvent('drag', {dragging:self, event:e});
737 
738 		//计算当前元素的具体位置坐标
739 		var selfPos = self.position();
740 		var draggingCoordinates = {
741 			top: selfPos.y,
742 			left: selfPos.x,
743 			right: selfPos.x + parseInt(self.getStyle('width')),
744 			bottom: selfPos.y + parseInt(self.getStyle('height'))
745 		}
746 
747 		//针对每一个容器，检查当前元素是否在容器当中
748 		for(var i=0,current,currentPos,containerCoordinates,l=self.__droppables.length; i<l; i++) {
749 			current = self.__droppables[i];
750 
751 			//计算每一个容器的边界
752 			currentPos = current.position();
753 			containerCoordinates = {
754 				top: currentPos.y,
755 				left: currentPos.x,
756 				right: currentPos.x + parseInt(current.getStyle('width')),
757 				bottom: currentPos.y + parseInt(current.getStyle('height'))
758 			}
759 			
760 			//判断容器的关系
761 			if(current == self.__belongToDroppable) {
762 				//如果容器是拖拽元素所属容器
763 				if(isInContainer(containerCoordinates, draggingCoordinates)) {
764 					//如果还在容器内，说明在所属容器内部移动，触发dragover事件
765 					current.fireEvent('dragover', {from:current, to:current, dragging:self});
766 				} else {
767 					//如果不在容器内，说明从所属容器中移出，触发dragleave事件
768 					current.fireEvent('dragleave', {from:current, to:null, dragging:self});
769 					self.__belongToDroppable = null;
770 				}
771 			//如果容器不是拖拽元素所属容器
772 			} else if(isInContainer(containerCoordinates, draggingCoordinates)) {
773 				//如果拖拽元素所属容器不为空，说明从拖拽容器中脱离出来了(是不是会跟上面事件触发有重复?试验还没出现这种情况)
774 				if(self.__belongToDroppable) {
775 					self.__belongToDroppable.fireEvent('dragleave', {from:self.__belongToDroppable, to:current, dragging:self});
776 				}
777 				//进入此容器了，触发dragenter
778 				//注意元素初始情况下会属于某个容器，初始化的时候要记录，避免错误的触发dragenter，mootools貌似没有判断
779 				current.fireEvent('dragenter', {from:self.__belongToDroppable, to:current, dragging:self});
780 				self.__belongToDroppable = current;
781 			}
782 		}	
783 	}
