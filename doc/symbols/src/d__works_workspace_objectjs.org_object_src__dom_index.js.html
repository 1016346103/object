 74 /**
 75  * 包装一个元素，使其拥有相应的Element包装成员
 76  * 比如 div 会使用 Element 进行包装
 77  * form 会使用 FormElement 进行包装
 78  * input / select 等会使用 FormItemElement 进行包装
 79  * 包装后的节点成员请参照相应的包装类成员
 80  * @function
 81  * @name dom.wrap
 82  * @param node 一个原生节点
 83  */
 84 var wrap = this.wrap = function(node) {
 85 	if (!node) return null;
 86 
 87 	if (Array.isArray(node)) {
 88 		return new Elements(node);
 89 	} else {
 90 		// 已经wrap过了
 91 		if (node._wrapped) return node;
 92 
 93 		var wrapper;
 94 		if (node === window) {
 95 			wrapper = Window;
 96 		} else if (node === window.document) {
 97 			wrapper = Document;
 98 		} else if (node.nodeType === 1) {
 99 			wrapper = getWrapper(node.tagName);
100 		} else {
101 			return node;
102 		}
103 
104 		// 尽早的设置_wrapped，因为在wrapper的initialize中可能出现递归调用（FormElement/FormItemElement）
105 		node._wrapped = true;
106 
107 		$uid(node);
108 
109 		Class.inject(wrapper, node);
110 
111 		return node;
112 	}
113 };

115 /**
116  * 通过selector获取context作用域下的节点集合
117  * dom.Elements包装后的节点数组拥有相应最小Element的统一调用方法
118  * 比如 forms = dom.getElements('form'); 'send' in forms // true
119  * @function
120  * @name dom.getElements
121  * @param selector 一个css selector
122  * @param context 一个节点
123  * @returns {dom.Elements}
124  */
125 var getElements = this.getElements = function(selector, context) {
126 	if (!context) context = document;
127 
128 	// 解析成Slick Selector对象
129 	var parsed = Slick.parse(selector);
130 
131 	// Slick在面对自定义标签时各种不靠谱，换用sizzle
132 	var eles = Sizzle(selector, context);
133 
134 	// 这里通过分析selector的最后一个部分的tagName，来确定这批eles的wrapper
135 	// 例如selector是 div form.xxx 则wrapper是 FormElement
136 	// 例如selector是 div .xxx 则wrapper是 Element
137 	// 例如selector是 div select.xxx, div.input.xxx 则wrapper是 FormItemElement
138 
139 	var wrapper, part;
140 	// 绝大部分情况都是length=0，只有1个selector，保证其性能
141 	if (parsed.expressions.length == 1) {
142 		part = parsed.expressions[0];
143 		wrapper = getWrapper(part[part.length - 1].tag);
144 
145 	// 由多个selector组成，比如 div select.xxx, div.input.xxx，要保证这种能取到 FormItemElement
146 	} else {
147 		// 通过生成每个selector wrapper的继承链，不断的生成当前selector和上一个selector的继承链的相同部分
148 		// 最后的chain的最后一个元素，既是公用wrapper
149 		for (var i = 0, chain, previousChain; i < parsed.expressions.length; i++) {
150 			part = parsed.expressions[i];
151 			wrapper = getWrapper(part[part.length - 1].tag);
152 
153 			// 当前selector最后元素的wrapper chain
154 			// slice(0, -1) 过滤掉Element继承的 Attribute 类
155 			chain = Class.getChain(wrapper).slice(0, -1).reverse();
156 			if (previousChain) {
157 				chain = getCommon(chain, previousChain);
158 			}
159 			// 如果相同部分length=1，则代表找到Element类了，可以停止继续搜索
160 			if (chain.length == 1) break;
161 			previousChain = chain;
162 		}
163 		wrapper = chain[chain.length - 1];
164 	}
165 
166 	return new Elements(eles, wrapper);
167 };
